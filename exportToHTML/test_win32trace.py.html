25       win32trace.write("Hi")
26       time.sleep(0.05)
27       if win32trace.read() != "Hi":
28           # Reset everything so following tests still fail with this error!
29           win32trace.TermRead()

33               "running - please stop this process and try again"

77       def testTermSematics(self):
78           win32trace.InitWrite()
79           win32trace.write("Ta da")
80   
81           # if we both Write and Read are terminated at the same time,
82           # we lose the data as the win32 object is closed.  Note that
83           # if another writer is running, we do *not* lose the data - so
84           # test for either the correct data or an empty string
85           win32trace.TermWrite()
86           win32trace.InitRead()
87           self.assertTrue(win32trace.read() in ("Ta da", ""))
88           win32trace.TermRead()
89   
90           # we keep the data because we init read before terminating write
91           win32trace.InitWrite()
92           win32trace.write("Ta da")
93           win32trace.InitRead()
94           win32trace.TermWrite()
95           self.assertEqual("Ta da", win32trace.read())

167  class WriterThread(threading.Thread):
168      def run(self):
169          self.writeCount = 0
170          for each in range(self.BucketCount):
171              win32trace.write(str(each))
172          self.writeCount = self.BucketCount
173  
174      def verifyWritten(self):
175          return self.writeCount == self.BucketCount

178  class TestMultipleThreadsWriting(unittest.TestCase):
179      # FullBucket is the thread count
180      FullBucket = 50
181      BucketCount = 9  # buckets must be a single digit number (ie. less than 10)
182  
183      def setUp(self):
184          SkipIfCI()
185          WriterThread.BucketCount = self.BucketCount
186          win32trace.InitRead()
187          win32trace.read()  # clear any old data.
188          win32trace.InitWrite()
189          CheckNoOtherReaders()
190          self.threads = [WriterThread() for each in range(self.FullBucket)]
191          self.buckets = list(range(self.BucketCount))
192          for each in self.buckets:
193              self.buckets[each] = 0
194  
195      def tearDown(self):
196          win32trace.TermRead()
197          win32trace.TermWrite()
198  
199      def areBucketsFull(self):
200          bucketsAreFull = True
201          for each in self.buckets:
202              assert each <= self.FullBucket, each
203              if each != self.FullBucket:
204                  bucketsAreFull = False
205                  break
206          return bucketsAreFull
207  
208      def read(self):
209          while 1:
210              readString = win32trace.blockingread()
211              for ch in readString:
212                  integer = int(ch)
213                  count = self.buckets[integer]
214                  assert count != -1
215                  self.buckets[integer] = count + 1
216                  if self.buckets[integer] == self.FullBucket:
217                      if self.areBucketsFull():
218                          return
219  
220      def testThreads(self):
221          for each in self.threads:
222              each.start()
223          self.read()
224          for each in self.threads:
225              each.join()
226          for each in self.threads:
227              assert each.verifyWritten()
228          assert self.areBucketsFull()

257  class TraceWriteProcess:
258      def __init__(self, threadCount):
259          self.exitCode = -1
260          self.threadCount = threadCount
261  
262      def start(self):
263          procHandle, threadHandle, procId, threadId = win32process.CreateProcess(
264              None,  # appName
265              'python.exe "%s" /run_test_process %s %s'
266              % (this_file, self.BucketCount, self.threadCount),
267              None,  # process security
268              None,  # thread security
269              0,  # inherit handles
270              win32process.NORMAL_PRIORITY_CLASS,
271              None,  # new environment
272              None,  # Current directory
273              win32process.STARTUPINFO(),  # startup info
274          )
275          self.processHandle = procHandle
276  
277      def join(self):
278          win32event.WaitForSingleObject(self.processHandle, win32event.INFINITE)
279          self.exitCode = win32process.GetExitCodeProcess(self.processHandle)
280  
281      def verifyWritten(self):
282          return self.exitCode == 0

342  def _RunAsTestProcess():
343      # Run as an external process by the main tests.
344      WriterThread.BucketCount = int(sys.argv[2])
345      threadCount = int(sys.argv[3])
346      threads = [WriterThread() for each in range(threadCount)]
347      win32trace.InitWrite()
348      for t in threads:
349          t.start()
350      for t in threads:
351          t.join()
352      for t in threads:
353          if not t.verifyWritten():
354              sys.exit(-1)
