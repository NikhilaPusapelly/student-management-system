8    def ReadLog(computer, logType="Application", dumpEachRecord=0):
9        # read the entire log back.
10       h = win32evtlog.OpenEventLog(computer, logType)
11       numRecords = win32evtlog.GetNumberOfEventLogRecords(h)
12       #       print "There are %d records" % numRecords
13   
14       num = 0
15       while 1:
16           objects = win32evtlog.ReadEventLog(
17               h,
18               win32evtlog.EVENTLOG_BACKWARDS_READ | win32evtlog.EVENTLOG_SEQUENTIAL_READ,
19               0,
20           )
21           if not objects:
22               break
23           for object in objects:
24               # get it for testing purposes, but dont print it.
25               msg = win32evtlogutil.SafeFormatMessage(object, logType)
26               if object.Sid is not None:
27                   try:
28                       domain, user, typ = win32security.LookupAccountSid(
29                           computer, object.Sid
30                       )
31                       sidDesc = "%s/%s" % (domain, user)
32                   except win32security.error:
33                       sidDesc = str(object.Sid)
34                   user_desc = "Event associated with user %s" % (sidDesc,)
35               else:
36                   user_desc = None
37               if dumpEachRecord:
38                   print(
39                       "Event record from %r generated at %s"
40                       % (object.SourceName, object.TimeGenerated.Format())
41                   )
42                   if user_desc:
43                       print(user_desc)
44                   try:
45                       print(msg)
46                   except UnicodeError:
47                       print("(unicode error printing message: repr() follows...)")
48                       print(repr(msg))
49   
50           num = num + len(objects)
51   
52       if numRecords == num:
53           print("Successfully read all", numRecords, "records")
54       else:
55           print(
56               "Couldn't get all records - reported %d, but found %d" % (numRecords, num)
57           )
58           print(
59               "(Note that some other app may have written records while we were running!)"
60           )
61       win32evtlog.CloseEventLog(h)

73   def test():
74       # check if running on Windows NT, if not, display notice and terminate
75       if win32api.GetVersion() & 0x80000000:
76           print("This sample only runs on NT")
77           return
78   
79       import getopt
80       import sys
81   
82       opts, args = getopt.getopt(sys.argv[1:], "rwh?c:t:v")
83       computer = None
84       do_read = do_write = 1
85   
86       logType = "Application"
87       verbose = 0
88   
89       if len(args) > 0:
90           print("Invalid args")
91           usage()
92           return 1
93       for opt, val in opts:
94           if opt == "-t":
95               logType = val
96           if opt == "-c":
97               computer = val
98           if opt in ["-h", "-?"]:
99               usage()
100              return
101          if opt == "-r":
102              do_read = 0
103          if opt == "-w":
104              do_write = 0
105          if opt == "-v":
106              verbose = verbose + 1
107      if do_write:
108          ph = win32api.GetCurrentProcess()
109          th = win32security.OpenProcessToken(ph, win32con.TOKEN_READ)
110          my_sid = win32security.GetTokenInformation(th, win32security.TokenUser)[0]
111  
112          win32evtlogutil.ReportEvent(
113              logType,
114              2,
115              strings=["The message text for event 2", "Another insert"],
116              data="Raw\0Data".encode("ascii"),
117              sid=my_sid,
118          )
119          win32evtlogutil.ReportEvent(
120              logType,
121              1,
122              eventType=win32evtlog.EVENTLOG_WARNING_TYPE,
123              strings=["A warning", "An even more dire warning"],
124              data="Raw\0Data".encode("ascii"),
125              sid=my_sid,
126          )
127          win32evtlogutil.ReportEvent(
128              logType,
129              1,
130              eventType=win32evtlog.EVENTLOG_INFORMATION_TYPE,
131              strings=["An info", "Too much info"],
132              data="Raw\0Data".encode("ascii"),
133              sid=my_sid,
134          )
135          print("Successfully wrote 3 records to the log")
136  
137      if do_read:
138          ReadLog(computer, logType, verbose > 0)
139
