8    def ParseArgs(str):
9        import string
10   
11       ret = []
12       pos = 0
13       length = len(str)
14       while pos < length:
15           try:
16               while str[pos] in string.whitespace:
17                   pos = pos + 1
18           except IndexError:
19               break
20           if pos >= length:
21               break
22           if str[pos] == '"':
23               pos = pos + 1
24               try:
25                   endPos = str.index('"', pos) - 1
26                   nextPos = endPos + 2
27               except ValueError:
28                   endPos = length
29                   nextPos = endPos + 1
30           else:
31               endPos = pos
32               while endPos < length and not str[endPos] in string.whitespace:
33                   endPos = endPos + 1
34               nextPos = endPos + 1
35           ret.append(str[pos : endPos + 1].strip())
36           pos = nextPos
37       return ret

40   def FixArgFileName(fileName):
41       """Convert a filename on the commandline to something useful. 
42       Given an automatic filename on the commandline, turn it a python module name, 
43       with the path added to sys.path."""
44       import os
45   
46       path, fname = os.path.split(fileName)
47       if len(path) == 0:
48           path = os.curdir
49       path = os.path.abspath(path)
50       # must check that the command line arg's path is in sys.path
51       for syspath in sys.path:
52           if os.path.abspath(syspath) == path:
53               break
54       else:
55           sys.path.append(path)
56       return os.path.splitext(fname)[0]
