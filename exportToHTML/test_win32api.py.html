79       def testValues(self):
80           key_name = r"PythonTestHarness\win32api"
81           ## tuples containing value name, value type, data
82           values = (
83               (None, win32con.REG_SZ, "This is default unnamed value"),
84               ("REG_SZ", win32con.REG_SZ, "REG_SZ text data"),
85               ("REG_EXPAND_SZ", win32con.REG_EXPAND_SZ, "%systemdir%"),
86               ## REG_MULTI_SZ value needs to be a list since strings are returned as a list
87               (
88                   "REG_MULTI_SZ",
89                   win32con.REG_MULTI_SZ,
90                   ["string 1", "string 2", "string 3", "string 4"],
91               ),
92               ("REG_MULTI_SZ_empty", win32con.REG_MULTI_SZ, []),
93               ("REG_DWORD", win32con.REG_DWORD, 666),
94               ("REG_QWORD_INT", win32con.REG_QWORD, 99),
95               ("REG_QWORD", win32con.REG_QWORD, 2**33),
96               (
97                   "REG_BINARY",
98                   win32con.REG_BINARY,
99                   str2bytes("\x00\x01\x02\x03\x04\x05\x06\x07\x08\x01\x00"),
100              ),
101          )
102  
103          hkey = win32api.RegCreateKey(win32con.HKEY_CURRENT_USER, key_name)
104          for value_name, reg_type, data in values:
105              win32api.RegSetValueEx(hkey, value_name, None, reg_type, data)
106  
107          for value_name, orig_type, orig_data in values:
108              data, typ = win32api.RegQueryValueEx(hkey, value_name)
109              self.assertEqual(typ, orig_type)
110              self.assertEqual(data, orig_data)

112      def testNotifyChange(self):
113          def change():
114              hkey = win32api.RegCreateKey(win32con.HKEY_CURRENT_USER, self.key_name)
115              try:
116                  win32api.RegSetValue(hkey, None, win32con.REG_SZ, "foo")
117              finally:
118                  win32api.RegDeleteKey(win32con.HKEY_CURRENT_USER, self.key_name)
119  
120          evt = win32event.CreateEvent(None, 0, 0, None)
121          ## REG_NOTIFY_CHANGE_LAST_SET - values
122          ## REG_CHANGE_NOTIFY_NAME - keys
123          ## REG_NOTIFY_CHANGE_SECURITY - security descriptor
124          ## REG_NOTIFY_CHANGE_ATTRIBUTES
125          win32api.RegNotifyChangeKeyValue(
126              win32con.HKEY_CURRENT_USER,
127              1,
128              win32api.REG_NOTIFY_CHANGE_LAST_SET,
129              evt,
130              True,
131          )
132          ret_code = win32event.WaitForSingleObject(evt, 0)
133          # Should be no change.
134          self.assertTrue(ret_code == win32con.WAIT_TIMEOUT)
135          change()
136          # Our event should now be in a signalled state.
137          ret_code = win32event.WaitForSingleObject(evt, 0)
138          self.assertTrue(ret_code == win32con.WAIT_OBJECT_0)

166          try:

168          except NameError:

180          long_name = win32api.GetLongPathNameW(short_name).lower()
181          self.assertTrue(

190      def testLongLongPathNames(self):
191          # We need filename where the FQN is > 256 - simplest way is to create a
192          # 250 character directory in the cwd (except - cwd may be on a drive
193          # not supporting \\\\?\\ (eg, network share) - so use temp.
194          import win32file
195  
196          basename = "a" * 250
197          # but we need to ensure we use the 'long' version of the
198          # temp dir for later comparison.
199          long_temp_dir = win32api.GetLongPathNameW(tempfile.gettempdir())
200          fname = "\\\\?\\" + os.path.join(long_temp_dir, basename)
201          try:
202              win32file.CreateDirectoryW(fname, None)
203          except win32api.error as details:
204              if details.winerror != winerror.ERROR_ALREADY_EXISTS:
205                  raise
206          try:
207              # GetFileAttributes automatically calls GetFileAttributesW when
208              # passed unicode
209              try:
210                  attr = win32api.GetFileAttributes(fname)
211              except win32api.error as details:
212                  if details.winerror != winerror.ERROR_FILENAME_EXCED_RANGE:
213                      raise
214  
215              attr = win32api.GetFileAttributes(str(fname))
216              self.assertTrue(attr & win32con.FILE_ATTRIBUTE_DIRECTORY, attr)
217  
218              long_name = win32api.GetLongPathNameW(fname)
219              self.assertEqual(long_name.lower(), fname.lower())
220          finally:
221              win32file.RemoveDirectory(fname)
