4    import os
5    import sys
6    
7    import pythoncom
8    import win32com
9    import win32com.client.makepy
10   import win32com.test
11   
12   genList = [
13       ("msword8", "{00020905-0000-0000-C000-000000000046}", 1033, 8, 0),
14   ]
15   
16   genDir = "Generated4Test"
17   
18   
19   def GetGenPath():
20       import win32api
21   
22       return os.path.join(win32api.GetFullPathName(win32com.test.__path__[0]), genDir)
23   
24   
25   def GenerateFromRegistered(fname, *loadArgs):
26       #       tlb = apply(pythoncom.LoadRegTypeLib, loadArgs)
27       genPath = GetGenPath()
28       try:
29           os.stat(genPath)
30       except os.error:
31           os.mkdir(genPath)
32       # Ensure an __init__ exists.
33       open(os.path.join(genPath, "__init__.py"), "w").close()
34       print(fname, ": generating -", end=" ")
35       f = open(os.path.join(genPath, fname + ".py"), "w")
36       win32com.client.makepy.GenerateFromTypeLibSpec(
37           loadArgs, f, bQuiet=1, bGUIProgress=1
38       )
39       f.close()
40       print("compiling -", end=" ")
41       fullModName = "win32com.test.%s.%s" % (genDir, fname)
42       exec("import " + fullModName)
43       # Inject the generated module as a top level module.
44       sys.modules[fname] = sys.modules[fullModName]
45       print("done")
46   
47   
48   def GenerateAll():
49       for args in genList:
50           try:
51               GenerateFromRegistered(*args)
52           except KeyboardInterrupt:
53               print("** Interrupted ***")
54               break
55           except pythoncom.com_error:
56               print("** Could not generate test code for ", args[0])
57   
58   
59   def CleanAll():
60       print("Cleaning generated test scripts...")
61       try:  # Clear exceptions!
62           1 / 0
63       except:
64           pass
65       genPath = GetGenPath()
66       for args in genList:
67           try:
68               name = args[0] + ".py"
69               os.unlink(os.path.join(genPath, name))
70           except os.error as details:
71               if type(details) == type(()) and details[0] != 2:
72                   print("Could not deleted generated", name, details)
73           try:
74               name = args[0] + ".pyc"
75               os.unlink(os.path.join(genPath, name))
76           except os.error as details:
77               if type(details) == type(()) and details[0] != 2:
78                   print("Could not deleted generated", name, details)
79           try:
80               os.unlink(os.path.join(genPath, "__init__.py"))
81           except:
82               pass
83           try:
84               os.unlink(os.path.join(genPath, "__init__.pyc"))
85           except:
86               pass
87       try:
88           os.rmdir(genPath)
89       except os.error as details:
90           print("Could not delete test directory -", details)
91   
92   
93   if __name__ == "__main__":
94       GenerateAll()
95       CleanAll()
