2    # bulkstamp.py:
3    #    Stamp versions on all files that can be found in a given tree.
4    #
5    # USAGE: python bulkstamp.py <version> <root directory> <descriptions>
6    #
7    # Example: python bulkstamp.py 103 ..\win32\Build\ desc.txt
8    #
9    # <version> corresponds to the build number. It will be concatenated with
10   # the major and minor version numbers found in the description file.
11   #
12   # Description information is pulled from an input text file with lines of
13   # the form:
14   #
15   #    <basename> <white space> <description>
16   #
17   # For example:
18   #
19   #    PyWinTypes.dll Common types for Python on Win32
20   #    etc
21   #
22   # The product's name, major, and minor versions are specified as:
23   #
24   #    name <white space> <value>
25   #    major <white space> <value>
26   #    minor <white space> <value>
27   #
28   # The tags are case-sensitive.
29   #
30   # Any line beginning with "#" will be ignored. Empty lines are okay.
31   #
32   
33   import fnmatch
34   import os

40   numStamped = 0
41   
42   g_patterns = [
43       "*.dll",
44       "*.pyd",
45       "*.exe",
46       "*.ocx",
47   ]

50   def walk(arg, dirname, names):
51       global numStamped
52       vars, debug, descriptions = arg
53       for name in names:
54           for pat in g_patterns:
55               if fnmatch.fnmatch(name, pat):
56                   # Handle the "_d" thing.
57                   pathname = os.path.join(dirname, name)
58                   base, ext = os.path.splitext(name)
59                   if base[-2:] == "_d":
60                       name = base[:-2] + ext
61                   is_dll = ext.lower() != ".exe"
62                   if os.path.normcase(name) in descriptions:
63                       desc = descriptions[os.path.normcase(name)]
64                       try:
65                           verstamp.stamp(vars, pathname, desc, is_dll=is_dll)
66                           numStamped = numStamped + 1
67                       except win32api.error as exc:
68                           print(
69                               "Could not stamp",
70                               pathname,
71                               "Error",
72                               exc.winerror,
73                               "-",
74                               exc.strerror,
75                           )
76                   else:
77                       print("WARNING: description not provided for:", name)
78                       # skip branding this - assume already branded or handled elsewhere
79

84   def load_descriptions(fname, vars):
85       retvars = {}
86       descriptions = {}
87   
88       lines = open(fname, "r").readlines()
89   
90       for i in range(len(lines)):
91           line = lines[i].strip()
92           if line != "" and line[0] != "#":
93               idx1 = line.find(" ")
94               idx2 = line.find("\t")
95               if idx1 == -1 or idx2 < idx1:
96                   idx1 = idx2
97               if idx1 == -1:
98                   print("ERROR: bad syntax in description file at line %d." % (i + 1))
99                   sys.exit(1)
100  
101              key = line[:idx1]
102              val = line[idx1:].strip()
103              if key in vars:
104                  retvars[key] = val
105              else:
106                  descriptions[key] = val
107  
108      if "product" not in retvars:
109          print("ERROR: description file is missing the product name.")
110          sys.exit(1)
111      if "major" not in retvars:
112          print("ERROR: description file is missing the major version number.")
113          sys.exit(1)
114      if "minor" not in retvars:
115          print("ERROR: description file is missing the minor version number.")
116          sys.exit(1)
117  
118      return retvars, descriptions

121  def scan(build, root, desc, **custom_vars):
122      global numStamped
123      numStamped = 0
124      try:
125          build = int(build)
126      except ValueError:
127          print("ERROR: build number is not a number: %s" % build)
128          sys.exit(1)
129  
130      debug = 0  ### maybe fix this one day
131  
132      varList = ["major", "minor", "sub", "company", "copyright", "trademarks", "product"]
133  
134      vars, descriptions = load_descriptions(desc, varList)
135      vars["build"] = build
136      vars.update(custom_vars)
137  
138      arg = vars, debug, descriptions
139      os.path.walk(root, walk, arg)
140  
141      print("Stamped %d files." % (numStamped))
142  
143  
144  if __name__ == "__main__":
145      if len(sys.argv) != 4:
146          print("ERROR: incorrect invocation. See script's header comments.")
147          sys.exit(1)
148  
149      scan(*tuple(sys.argv[1:]))
