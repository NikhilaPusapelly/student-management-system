14   class PipeTests(unittest.TestCase):
15       pipename = "\\\\.\\pipe\\python_test_pipe"

17       def _serverThread(self, pipe_handle, event, wait_time):
18           # just do one connection and terminate.
19           hr = win32pipe.ConnectNamedPipe(pipe_handle)
20           self.assertTrue(
21               hr in (0, winerror.ERROR_PIPE_CONNECTED), "Got error code 0x%x" % (hr,)
22           )
23           hr, got = win32file.ReadFile(pipe_handle, 100)
24           self.assertEqual(got, str2bytes("foo\0bar"))
25           time.sleep(wait_time)
26           win32file.WriteFile(pipe_handle, str2bytes("bar\0foo"))
27           pipe_handle.Close()
28           event.set()
29   
30       def startPipeServer(self, event, wait_time=0):
31           openMode = win32pipe.PIPE_ACCESS_DUPLEX
32           pipeMode = win32pipe.PIPE_TYPE_MESSAGE | win32pipe.PIPE_WAIT
33   
34           sa = pywintypes.SECURITY_ATTRIBUTES()
35           sa.SetSecurityDescriptorDacl(1, None, 0)
36   
37           pipe_handle = win32pipe.CreateNamedPipe(
38               self.pipename,
39               openMode,
40               pipeMode,
41               win32pipe.PIPE_UNLIMITED_INSTANCES,
42               0,
43               0,
44               2000,
45               sa,
46           )
47   
48           threading.Thread(
49               target=self._serverThread, args=(pipe_handle, event, wait_time)
50           ).start()
51   
52       def testCallNamedPipe(self):
53           event = threading.Event()
54           self.startPipeServer(event)
55   
56           got = win32pipe.CallNamedPipe(
57               self.pipename, str2bytes("foo\0bar"), 1024, win32pipe.NMPWAIT_WAIT_FOREVER
58           )
59           self.assertEqual(got, str2bytes("bar\0foo"))
60           event.wait(5)
61           self.assertTrue(event.isSet(), "Pipe server thread didn't terminate")
62   
63       def testTransactNamedPipeBlocking(self):
64           event = threading.Event()
65           self.startPipeServer(event)
66           open_mode = win32con.GENERIC_READ | win32con.GENERIC_WRITE
67   
68           hpipe = win32file.CreateFile(
69               self.pipename,
70               open_mode,
71               0,  # no sharing
72               None,  # default security
73               win32con.OPEN_EXISTING,
74               0,  # win32con.FILE_FLAG_OVERLAPPED,
75               None,
76           )
77   
78           # set to message mode.
79           win32pipe.SetNamedPipeHandleState(
80               hpipe, win32pipe.PIPE_READMODE_MESSAGE, None, None
81           )
82   
83           hr, got = win32pipe.TransactNamedPipe(hpipe, str2bytes("foo\0bar"), 1024, None)
84           self.assertEqual(got, str2bytes("bar\0foo"))
85           event.wait(5)
86           self.assertTrue(event.isSet(), "Pipe server thread didn't terminate")

118      def testTransactNamedPipeAsync(self):
119          event = threading.Event()
120          overlapped = pywintypes.OVERLAPPED()
121          overlapped.hEvent = win32event.CreateEvent(None, 0, 0, None)
122          self.startPipeServer(event, 0.5)
123          open_mode = win32con.GENERIC_READ | win32con.GENERIC_WRITE
124  
125          hpipe = win32file.CreateFile(
126              self.pipename,
127              open_mode,
128              0,  # no sharing
129              None,  # default security
130              win32con.OPEN_EXISTING,
131              win32con.FILE_FLAG_OVERLAPPED,
132              None,
133          )
134  
135          # set to message mode.
136          win32pipe.SetNamedPipeHandleState(
137              hpipe, win32pipe.PIPE_READMODE_MESSAGE, None, None
138          )
139  
140          buffer = win32file.AllocateReadBuffer(1024)
141          hr, got = win32pipe.TransactNamedPipe(
142              hpipe, str2bytes("foo\0bar"), buffer, overlapped
143          )
144          self.assertEqual(hr, winerror.ERROR_IO_PENDING)
145          nbytes = win32file.GetOverlappedResult(hpipe, overlapped, True)
146          got = buffer[:nbytes]
147          self.assertEqual(got, str2bytes("bar\0foo"))
148          event.wait(5)
149          self.assertTrue(event.isSet(), "Pipe server thread didn't terminate")
