33   def ApplyIgnoreError(fn, args):
34       try:
35           return fn(*args)
36       except error:  # Ignore win32api errors.
37           return None

40   class TestPipeService(win32serviceutil.ServiceFramework):
41       _svc_name_ = "PyPipeTestService"
42       _svc_display_name_ = "Python Pipe Test Service"
43       _svc_description_ = "Tests Python service framework by receiving and echoing messages over a named pipe"
44   
45       def __init__(self, args):
46           win32serviceutil.ServiceFramework.__init__(self, args)
47           self.hWaitStop = CreateEvent(None, 0, 0, None)
48           self.overlapped = pywintypes.OVERLAPPED()
49           self.overlapped.hEvent = CreateEvent(None, 0, 0, None)
50           self.thread_handles = []
51   
52       def CreatePipeSecurityObject(self):
53           # Create a security object giving World read/write access,
54           # but only "Owner" modify access.
55           sa = pywintypes.SECURITY_ATTRIBUTES()
56           sidEveryone = pywintypes.SID()
57           sidEveryone.Initialize(SECURITY_WORLD_SID_AUTHORITY, 1)
58           sidEveryone.SetSubAuthority(0, SECURITY_WORLD_RID)
59           sidCreator = pywintypes.SID()
60           sidCreator.Initialize(SECURITY_CREATOR_SID_AUTHORITY, 1)
61           sidCreator.SetSubAuthority(0, SECURITY_CREATOR_OWNER_RID)
62   
63           acl = pywintypes.ACL()
64           acl.AddAccessAllowedAce(FILE_GENERIC_READ | FILE_GENERIC_WRITE, sidEveryone)
65           acl.AddAccessAllowedAce(FILE_ALL_ACCESS, sidCreator)
66   
67           sa.SetSecurityDescriptorDacl(1, acl, 0)
68           return sa

70       # The functions executed in their own thread to process a client request.
71       def DoProcessClient(self, pipeHandle, tid):
72           try:
73               try:
74                   # Create a loop, reading large data.  If we knew the data stream was
75                   # was small, a simple ReadFile would do.
76                   d = "".encode("ascii")  # ensure bytes on py2k and py3k...
77                   hr = winerror.ERROR_MORE_DATA
78                   while hr == winerror.ERROR_MORE_DATA:
79                       hr, thisd = ReadFile(pipeHandle, 256)
80                       d = d + thisd
81                   print("Read", d)
82                   ok = 1
83               except error:
84                   # Client disconnection - do nothing
85                   ok = 0
86   
87               # A secure service would handle (and ignore!) errors writing to the
88               # pipe, but for the sake of this demo we dont (if only to see what errors
89               # we can get when our clients break at strange times :-)
90               if ok:
91                   msg = (
92                       "%s (on thread %d) sent me %s"
93                       % (GetNamedPipeHandleState(pipeHandle, False, True)[4], tid, d)
94                   ).encode("ascii")
95                   WriteFile(pipeHandle, msg)
96           finally:
97               ApplyIgnoreError(DisconnectNamedPipe, (pipeHandle,))
98               ApplyIgnoreError(CloseHandle, (pipeHandle,))
99   
100      def ProcessClient(self, pipeHandle):
101          try:
102              procHandle = GetCurrentProcess()
103              th = DuplicateHandle(
104                  procHandle,
105                  GetCurrentThread(),
106                  procHandle,
107                  0,
108                  0,
109                  win32con.DUPLICATE_SAME_ACCESS,
110              )
111              try:
112                  self.thread_handles.append(th)
113                  try:
114                      return self.DoProcessClient(pipeHandle, th)
115                  except:
116                      traceback.print_exc()
117              finally:
118                  self.thread_handles.remove(th)
119          except:
120              traceback.print_exc()

122      def SvcStop(self):
123          self.ReportServiceStatus(win32service.SERVICE_STOP_PENDING)
124          SetEvent(self.hWaitStop)

126      def SvcDoRun(self):
127          # Write an event log record - in debug mode we will also
128          # see this message printed.
129          servicemanager.LogMsg(
130              servicemanager.EVENTLOG_INFORMATION_TYPE,
131              servicemanager.PYS_SERVICE_STARTED,
132              (self._svc_name_, ""),
133          )
134  
135          num_connections = 0
136          while 1:
137              pipeHandle = CreateNamedPipe(
138                  "\\\\.\\pipe\\PyPipeTest",
139                  PIPE_ACCESS_DUPLEX | FILE_FLAG_OVERLAPPED,
140                  PIPE_TYPE_MESSAGE | PIPE_READMODE_BYTE,
141                  PIPE_UNLIMITED_INSTANCES,  # max instances
142                  0,
143                  0,
144                  6000,
145                  self.CreatePipeSecurityObject(),
146              )
147              try:
148                  hr = ConnectNamedPipe(pipeHandle, self.overlapped)
149              except error as details:
150                  print("Error connecting pipe!", details)
151                  CloseHandle(pipeHandle)
152                  break
153              if hr == winerror.ERROR_PIPE_CONNECTED:
154                  # Client is already connected - signal event
155                  SetEvent(self.overlapped.hEvent)
156              rc = WaitForMultipleObjects(
157                  (self.hWaitStop, self.overlapped.hEvent), 0, INFINITE
158              )
159              if rc == WAIT_OBJECT_0:
160                  # Stop event
161                  break
162              else:
163                  # Pipe event - spawn thread to deal with it.
164                  _thread.start_new_thread(self.ProcessClient, (pipeHandle,))
165                  num_connections = num_connections + 1
166  
167          # Sleep to ensure that any new threads are in the list, and then
168          # wait for all current threads to finish.
169          # What is a better way?
170          Sleep(500)
171          while self.thread_handles:
172              self.ReportServiceStatus(win32service.SERVICE_STOP_PENDING, 5000)
173              print("Waiting for %d threads to finish..." % (len(self.thread_handles)))
174              WaitForMultipleObjects(self.thread_handles, 1, 3000)
175          # Write another event log record.
176          servicemanager.LogMsg(
177              servicemanager.EVENTLOG_INFORMATION_TYPE,
178              servicemanager.PYS_SERVICE_STOPPED,
179              (self._svc_name_, " after processing %d connections" % (num_connections,)),
180          )
