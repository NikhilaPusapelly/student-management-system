8        if k.startswith("VK_"):

11   free_console = True
12   try:
13       win32console.AllocConsole()
14   except win32console.error as exc:
15       if exc.winerror != 5:
16           raise
17       ## only free console if one was created successfully
18       free_console = False
19   
20   stdout = win32console.GetStdHandle(win32console.STD_OUTPUT_HANDLE)
21   stdin = win32console.GetStdHandle(win32console.STD_INPUT_HANDLE)
22   newbuffer = win32console.CreateConsoleScreenBuffer()
23   newbuffer.SetConsoleActiveScreenBuffer()
24   newbuffer.SetConsoleTextAttribute(
25       win32console.FOREGROUND_RED
26       | win32console.FOREGROUND_INTENSITY
27       | win32console.BACKGROUND_GREEN
28       | win32console.BACKGROUND_INTENSITY
29   )
30   newbuffer.WriteConsole("This is a new screen buffer\n")
31   
32   ## test setting screen buffer and window size
33   ## screen buffer size cannot be smaller than window size
34   window_size = newbuffer.GetConsoleScreenBufferInfo()["Window"]
35   coord = win32console.PyCOORDType(X=window_size.Right + 20, Y=window_size.Bottom + 20)
36   newbuffer.SetConsoleScreenBufferSize(coord)
37   
38   window_size.Right += 10
39   window_size.Bottom += 10
40   newbuffer.SetConsoleWindowInfo(Absolute=True, ConsoleWindow=window_size)
41   
42   ## write some records to the input queue
43   x = win32console.PyINPUT_RECORDType(win32console.KEY_EVENT)
44   x.Char = "X"
45   x.KeyDown = True
46   x.RepeatCount = 1
47   x.VirtualKeyCode = 0x58
48   x.ControlKeyState = win32con.SHIFT_PRESSED
49   
50   z = win32console.PyINPUT_RECORDType(win32console.KEY_EVENT)
51   z.Char = "Z"
52   z.KeyDown = True
53   z.RepeatCount = 1
54   z.VirtualKeyCode = 0x5A
55   z.ControlKeyState = win32con.SHIFT_PRESSED
56   
57   stdin.WriteConsoleInput([x, z, x])
58   
59   newbuffer.SetConsoleTextAttribute(
60       win32console.FOREGROUND_RED
61       | win32console.FOREGROUND_INTENSITY
62       | win32console.BACKGROUND_GREEN
63       | win32console.BACKGROUND_INTENSITY
64   )
65   newbuffer.WriteConsole("Press some keys, click some characters with the mouse\n")
66   
67   newbuffer.SetConsoleTextAttribute(
68       win32console.FOREGROUND_BLUE
69       | win32console.FOREGROUND_INTENSITY
70       | win32console.BACKGROUND_RED
71       | win32console.BACKGROUND_INTENSITY
72   )
73   newbuffer.WriteConsole('Hit "End" key to quit\n')
74   
75   breakout = False
76   while not breakout:
77       input_records = stdin.ReadConsoleInput(10)
78       for input_record in input_records:
79           if input_record.EventType == win32console.KEY_EVENT:
80               if input_record.KeyDown:
81                   if input_record.Char == "\0":
82                       newbuffer.WriteConsole(
83                           virtual_keys.get(
84                               input_record.VirtualKeyCode,
85                               "VirtualKeyCode: %s" % input_record.VirtualKeyCode,
86                           )
87                       )
88                   else:
89                       newbuffer.WriteConsole(input_record.Char)
90                   if input_record.VirtualKeyCode == win32con.VK_END:
91                       breakout = True
92                       break
93           elif input_record.EventType == win32console.MOUSE_EVENT:
94               if input_record.EventFlags == 0:  ## 0 indicates a button event
95                   if input_record.ButtonState != 0:  ## exclude button releases
96                       pos = input_record.MousePosition
97                       # switch the foreground and background colors of the character that was clicked
98                       attr = newbuffer.ReadConsoleOutputAttribute(
99                           Length=1, ReadCoord=pos
100                      )[0]
101                      new_attr = attr
102                      if attr & win32console.FOREGROUND_BLUE:
103                          new_attr = (
104                              new_attr & ~win32console.FOREGROUND_BLUE
105                          ) | win32console.BACKGROUND_BLUE
106                      if attr & win32console.FOREGROUND_RED:
107                          new_attr = (
108                              new_attr & ~win32console.FOREGROUND_RED
109                          ) | win32console.BACKGROUND_RED
110                      if attr & win32console.FOREGROUND_GREEN:
111                          new_attr = (
112                              new_attr & ~win32console.FOREGROUND_GREEN
113                          ) | win32console.BACKGROUND_GREEN
114  
115                      if attr & win32console.BACKGROUND_BLUE:
116                          new_attr = (
117                              new_attr & ~win32console.BACKGROUND_BLUE
118                          ) | win32console.FOREGROUND_BLUE
119                      if attr & win32console.BACKGROUND_RED:
120                          new_attr = (
121                              new_attr & ~win32console.BACKGROUND_RED
122                          ) | win32console.FOREGROUND_RED
123                      if attr & win32console.BACKGROUND_GREEN:
124                          new_attr = (
125                              new_attr & ~win32console.BACKGROUND_GREEN
126                          ) | win32console.FOREGROUND_GREEN
127                      newbuffer.WriteConsoleOutputAttribute((new_attr,), pos)
128          else:
129              newbuffer.WriteConsole(str(input_record))
130      time.sleep(0.1)
131  
132  stdout.SetConsoleActiveScreenBuffer()
133  newbuffer.Close()
