26   def _accept(prefix):
27       is_riff_file_format = prefix[:4] == b"RIFF"
28       is_webp_file = prefix[8:12] == b"WEBP"
29       is_valid_vp8_mode = prefix[12:16] in _VP8_MODES_BY_IDENTIFIER
30   
31       if is_riff_file_format and is_webp_file and is_valid_vp8_mode:
32           if not SUPPORTED:
33               return (
34                   "image file could not be identified because WEBP support not installed"
35               )
36           return True

39   class WebPImageFile(ImageFile.ImageFile):
40       format = "WEBP"
41       format_description = "WebP image"
42       __loaded = 0
43       __logical_frame = 0
44   
45       def _open(self):
46           if not _webp.HAVE_WEBPANIM:
47               # Legacy mode
48               data, width, height, self._mode, icc_profile, exif = _webp.WebPDecode(
49                   self.fp.read()
50               )
51               if icc_profile:
52                   self.info["icc_profile"] = icc_profile
53               if exif:
54                   self.info["exif"] = exif
55               self._size = width, height
56               self.fp = BytesIO(data)
57               self.tile = [("raw", (0, 0) + self.size, 0, self.mode)]
58               self.n_frames = 1
59               self.is_animated = False
60               return
61   
62           # Use the newer AnimDecoder API to parse the (possibly) animated file,
63           # and access muxed chunks like ICC/EXIF/XMP.
64           self._decoder = _webp.WebPAnimDecoder(self.fp.read())
65   
66           # Get info from decoder
67           width, height, loop_count, bgcolor, frame_count, mode = self._decoder.get_info()
68           self._size = width, height
69           self.info["loop"] = loop_count
70           bg_a, bg_r, bg_g, bg_b = (
71               (bgcolor >> 24) & 0xFF,
72               (bgcolor >> 16) & 0xFF,
73               (bgcolor >> 8) & 0xFF,
74               bgcolor & 0xFF,
75           )
76           self.info["background"] = (bg_r, bg_g, bg_b, bg_a)
77           self.n_frames = frame_count
78           self.is_animated = self.n_frames > 1
79           self._mode = "RGB" if mode == "RGBX" else mode
80           self.rawmode = mode
81           self.tile = []
82   
83           # Attempt to read ICC / EXIF / XMP chunks from file
84           icc_profile = self._decoder.get_chunk("ICCP")
85           exif = self._decoder.get_chunk("EXIF")
86           xmp = self._decoder.get_chunk("XMP ")
87           if icc_profile:
88               self.info["icc_profile"] = icc_profile
89           if exif:
90               self.info["exif"] = exif
91           if xmp:
92               self.info["xmp"] = xmp
93   
94           # Initialize seek state
95           self._reset(reset=False)
96   
97       def _getexif(self):
98           if "exif" not in self.info:
99               return None
100          return self.getexif()._get_merged_dict()
101  
102      def getxmp(self):
103          """ 
104          Returns a dictionary containing the XMP tags. 
105          Requires defusedxml to be installed. 
106   
107          :returns: XMP tags in a dictionary. 
108          """
109          return self._getxmp(self.info["xmp"]) if "xmp" in self.info else {}

111      def seek(self, frame):
112          if not self._seek_check(frame):
113              return
114  
115          # Set logical frame to requested position
116          self.__logical_frame = frame
117  
118      def _reset(self, reset=True):
119          if reset:
120              self._decoder.reset()
121          self.__physical_frame = 0
122          self.__loaded = -1
123          self.__timestamp = 0
124  
125      def _get_next(self):
126          # Get next frame
127          ret = self._decoder.get_next()
128          self.__physical_frame += 1
129  
130          # Check if an error occurred
131          if ret is None:
132              self._reset()  # Reset just to be safe
133              self.seek(0)
134              msg = "failed to decode next frame in WebP file"
135              raise EOFError(msg)
136  
137          # Compute duration
138          data, timestamp = ret
139          duration = timestamp - self.__timestamp
140          self.__timestamp = timestamp
141  
142          # libwebp gives frame end, adjust to start of frame
143          timestamp -= duration
144          return data, timestamp, duration
145  
146      def _seek(self, frame):
147          if self.__physical_frame == frame:
148              return  # Nothing to do
149          if frame < self.__physical_frame:
150              self._reset()  # Rewind to beginning
151          while self.__physical_frame < frame:
152              self._get_next()  # Advance to the requested frame
153  
154      def load(self):
155          if _webp.HAVE_WEBPANIM:
156              if self.__loaded != self.__logical_frame:
157                  self._seek(self.__logical_frame)
158  
159                  # We need to load the image data for this frame
160                  data, timestamp, duration = self._get_next()
161                  self.info["timestamp"] = timestamp
162                  self.info["duration"] = duration
163                  self.__loaded = self.__logical_frame
164  
165                  # Set tile
166                  if self.fp and self._exclusive_fp:
167                      self.fp.close()
168                  self.fp = BytesIO(data)
169                  self.tile = [("raw", (0, 0) + self.size, 0, self.rawmode)]
170  
171          return super().load()
172  
173      def load_seek(self, pos):
174          pass
175  
176      def tell(self):
177          if not _webp.HAVE_WEBPANIM:
178              return super().tell()
179  
180          return self.__logical_frame

183  def _save_all(im, fp, filename):
184      encoderinfo = im.encoderinfo.copy()
185      append_images = list(encoderinfo.get("append_images", []))
186  
187      # If total frame count is 1, then save using the legacy API, which
188      # will preserve non-alpha modes
189      total = 0
190      for ims in [im] + append_images:
191          total += getattr(ims, "n_frames", 1)
192      if total == 1:
193          _save(im, fp, filename)
194          return
195  
196      background = (0, 0, 0, 0)
197      if "background" in encoderinfo:
198          background = encoderinfo["background"]
199      elif "background" in im.info:
200          background = im.info["background"]
201          if isinstance(background, int):
202              # GifImagePlugin stores a global color table index in
203              # info["background"]. So it must be converted to an RGBA value
204              palette = im.getpalette()
205              if palette:
206                  r, g, b = palette[background * 3 : (background + 1) * 3]
207                  background = (r, g, b, 255)
208              else:
209                  background = (background, background, background, 255)
210  
211      duration = im.encoderinfo.get("duration", im.info.get("duration", 0))
212      loop = im.encoderinfo.get("loop", 0)
213      minimize_size = im.encoderinfo.get("minimize_size", False)
214      kmin = im.encoderinfo.get("kmin", None)
215      kmax = im.encoderinfo.get("kmax", None)
216      allow_mixed = im.encoderinfo.get("allow_mixed", False)
217      verbose = False
218      lossless = im.encoderinfo.get("lossless", False)
219      quality = im.encoderinfo.get("quality", 80)
220      alpha_quality = im.encoderinfo.get("alpha_quality", 100)
221      method = im.encoderinfo.get("method", 0)
222      icc_profile = im.encoderinfo.get("icc_profile") or ""
223      exif = im.encoderinfo.get("exif", "")
224      if isinstance(exif, Image.Exif):
225          exif = exif.tobytes()
226      xmp = im.encoderinfo.get("xmp", "")
227      if allow_mixed:
228          lossless = False
229  
230      # Sensible keyframe defaults are from gif2webp.c script
231      if kmin is None:
232          kmin = 9 if lossless else 3
233      if kmax is None:
234          kmax = 17 if lossless else 5
235  
236      # Validate background color
237      if (
238          not isinstance(background, (list, tuple))
239          or len(background) != 4
240          or not all(0 <= v < 256 for v in background)
241      ):
242          msg = f"Background color is not an RGBA tuple clamped to (0-255): {background}"
243          raise OSError(msg)
244  
245      # Convert to packed uint
246      bg_r, bg_g, bg_b, bg_a = background
247      background = (bg_a << 24) | (bg_r << 16) | (bg_g << 8) | (bg_b << 0)
248  
249      # Setup the WebP animation encoder
250      enc = _webp.WebPAnimEncoder(
251          im.size[0],
252          im.size[1],
253          background,
254          loop,
255          minimize_size,
256          kmin,
257          kmax,
258          allow_mixed,
259          verbose,
260      )
261  
262      # Add each frame
263      frame_idx = 0
264      timestamp = 0
265      cur_idx = im.tell()
266      try:
267          for ims in [im] + append_images:
268              # Get # of frames in this image
269              nfr = getattr(ims, "n_frames", 1)
270  
271              for idx in range(nfr):
272                  ims.seek(idx)
273                  ims.load()
274  
275                  # Make sure image mode is supported
276                  frame = ims
277                  rawmode = ims.mode
278                  if ims.mode not in _VALID_WEBP_MODES:
279                      alpha = (
280                          "A" in ims.mode
281                          or "a" in ims.mode
282                          or (ims.mode == "P" and "A" in ims.im.getpalettemode())
283                      )
284                      rawmode = "RGBA" if alpha else "RGB"
285                      frame = ims.convert(rawmode)
286  
287                  if rawmode == "RGB":
288                      # For faster conversion, use RGBX
289                      rawmode = "RGBX"
290  
291                  # Append the frame to the animation encoder
292                  enc.add(
293                      frame.tobytes("raw", rawmode),
294                      round(timestamp),
295                      frame.size[0],
296                      frame.size[1],
297                      rawmode,
298                      lossless,
299                      quality,
300                      alpha_quality,
301                      method,
302                  )
303  
304                  # Update timestamp and frame index
305                  if isinstance(duration, (list, tuple)):
306                      timestamp += duration[frame_idx]
307                  else:
308                      timestamp += duration
309                  frame_idx += 1
310  
311      finally:
312          im.seek(cur_idx)
313  
314      # Force encoder to flush frames
315      enc.add(None, round(timestamp), 0, 0, "", lossless, quality, alpha_quality, 0)
316  
317      # Get the final output from the encoder
318      data = enc.assemble(icc_profile, exif, xmp)
319      if data is None:
320          msg = "cannot write file as WebP (encoder returned None)"
321          raise OSError(msg)
322  
323      fp.write(data)
324

326  def _save(im, fp, filename):
327      lossless = im.encoderinfo.get("lossless", False)
328      quality = im.encoderinfo.get("quality", 80)
329      alpha_quality = im.encoderinfo.get("alpha_quality", 100)
330      icc_profile = im.encoderinfo.get("icc_profile") or ""
331      exif = im.encoderinfo.get("exif", b"")
332      if isinstance(exif, Image.Exif):
333          exif = exif.tobytes()
334      if exif.startswith(b"Exif\x00\x00"):
335          exif = exif[6:]
336      xmp = im.encoderinfo.get("xmp", "")
337      method = im.encoderinfo.get("method", 4)
338      exact = 1 if im.encoderinfo.get("exact") else 0
339  
340      if im.mode not in _VALID_WEBP_LEGACY_MODES:
341          im = im.convert("RGBA" if im.has_transparency_data else "RGB")
342  
343      data = _webp.WebPEncode(
344          im.tobytes(),
345          im.size[0],
346          im.size[1],
347          lossless,
348          float(quality),
349          float(alpha_quality),
350          im.mode,
351          icc_profile,
352          method,
353          exact,
354          exif,
355          xmp,
356      )
357      if data is None:
358          msg = "cannot write file as WebP (encoder returned None)"
359          raise OSError(msg)
360  
361      fp.write(data)
