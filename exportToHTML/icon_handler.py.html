1    # A sample icon handler.  Sets the icon for Python files to a random
2    # ICO file.  ICO files are found in the Python directory - generally there will
3    # be 3 icons found.
4    #
5    # To demostrate:
6    # * Execute this script to register the context menu.
7    # * Open Windows Explorer, and browse to a directory with a .py file.
8    # * Note the pretty, random selection of icons!
9    # Use glob to locate ico files, and random.choice to pick one.
10   import glob
11   import os
12   import random
13   import sys
14   
15   import pythoncom
16   import win32gui
17   import winerror
18   from win32com.shell import shell, shellcon
19   
20   ico_files = glob.glob(os.path.join(sys.prefix, "*.ico"))
21   if not ico_files:
22       ico_files = glob.glob(os.path.join(sys.prefix, "PC", "*.ico"))
23   if not ico_files:
24       print("WARNING: Can't find any icon files")
25   
26   # Our shell extension.
27   IExtractIcon_Methods = "Extract GetIconLocation".split()
28   IPersistFile_Methods = "IsDirty Load Save SaveCompleted GetCurFile".split()
29   
30   
31   class ShellExtension:
32       _reg_progid_ = "Python.ShellExtension.IconHandler"
33       _reg_desc_ = "Python Sample Shell Extension (icon handler)"
34       _reg_clsid_ = "{a97e32d7-3b78-448c-b341-418120ea9227}"
35       _com_interfaces_ = [shell.IID_IExtractIcon, pythoncom.IID_IPersistFile]
36       _public_methods_ = IExtractIcon_Methods + IPersistFile_Methods
37   
38       def Load(self, filename, mode):
39           self.filename = filename
40           self.mode = mode
41   
42       def GetIconLocation(self, flags):
43           # note - returning a single int will set the HRESULT (eg, S_FALSE,
44           # E_PENDING - see MS docs for details.
45           return random.choice(ico_files), 0, 0
46   
47       def Extract(self, fname, index, size):
48           return winerror.S_FALSE
49   
50   
51   def DllRegisterServer():
52       import winreg
53   
54       key = winreg.CreateKey(winreg.HKEY_CLASSES_ROOT, "Python.File\\shellex")
55       subkey = winreg.CreateKey(key, "IconHandler")
56       winreg.SetValueEx(subkey, None, 0, winreg.REG_SZ, ShellExtension._reg_clsid_)
57       print(ShellExtension._reg_desc_, "registration complete.")
58   
59   
60   def DllUnregisterServer():
61       import winreg
62   
63       try:
64           key = winreg.DeleteKey(
65               winreg.HKEY_CLASSES_ROOT, "Python.File\\shellex\\IconHandler"
66           )
67       except WindowsError as details:
68           import errno
69   
70           if details.errno != errno.ENOENT:
71               raise
72       print(ShellExtension._reg_desc_, "unregistration complete.")
73   
74   
75   if __name__ == "__main__":
76       from win32com.server import register
77   
78       register.UseCommandLine(
79           ShellExtension,
80           finalize_register=DllRegisterServer,
81           finalize_unregister=DllUnregisterServer,
82       )
