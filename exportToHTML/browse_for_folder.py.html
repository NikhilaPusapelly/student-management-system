3    import os

5    import win32gui
6    from win32com.shell import shell, shellcon
7    
8    
9    # A callback procedure - called by SHBrowseForFolder
10   def BrowseCallbackProc(hwnd, msg, lp, data):
11       if msg == shellcon.BFFM_INITIALIZED:
12           win32gui.SendMessage(hwnd, shellcon.BFFM_SETSELECTION, 1, data)
13       elif msg == shellcon.BFFM_SELCHANGED:
14           # Set the status text of the
15           # For this message, 'lp' is the address of the PIDL.
16           pidl = shell.AddressAsPIDL(lp)
17           try:
18               path = shell.SHGetPathFromIDList(pidl)
19               win32gui.SendMessage(hwnd, shellcon.BFFM_SETSTATUSTEXT, 0, path)
20           except shell.error:
21               # No path for this PIDL
22               pass

25   if __name__ == "__main__":
26       # Demonstrate a dialog with the cwd selected as the default - this
27       # must be done via a callback function.
28       flags = shellcon.BIF_STATUSTEXT
29       shell.SHBrowseForFolder(
30           0,  # parent HWND
31           None,  # root PIDL.
32           "Default of %s" % os.getcwd(),  # title
33           flags,  # flags
34           BrowseCallbackProc,  # callback function
35           os.getcwd(),  # 'data' param for the callback
36       )
37       # Browse from this directory down only.
38       # Get the PIDL for the cwd.
39       desktop = shell.SHGetDesktopFolder()
40       cb, pidl, extra = desktop.ParseDisplayName(0, None, os.getcwd())
41       shell.SHBrowseForFolder(
42           0,  # parent HWND
43           pidl,  # root PIDL.
44           "From %s down only" % os.getcwd(),  # title
