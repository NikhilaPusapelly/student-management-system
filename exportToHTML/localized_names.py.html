1    # A Python port of the MS knowledge base article Q157234
2    # "How to deal with localized and renamed user and group names"
3    # http://support.microsoft.com/default.aspx?kbid=157234
4    
5    import sys
6    
7    import pywintypes
8    from ntsecuritycon import *
9    from win32net import NetUserModalsGet
10   from win32security import LookupAccountSid
11   
12   
13   def LookupAliasFromRid(TargetComputer, Rid):
14       # Sid is the same regardless of machine, since the well-known
15       # BUILTIN domain is referenced.
16       sid = pywintypes.SID()
17       sid.Initialize(SECURITY_NT_AUTHORITY, 2)
18   
19       for i, r in enumerate((SECURITY_BUILTIN_DOMAIN_RID, Rid)):
20           sid.SetSubAuthority(i, r)
21   
22       name, domain, typ = LookupAccountSid(TargetComputer, sid)
23       return name
24   
25   
26   def LookupUserGroupFromRid(TargetComputer, Rid):
27       # get the account domain Sid on the target machine
28       # note: if you were looking up multiple sids based on the same
29       # account domain, only need to call this once.
30       umi2 = NetUserModalsGet(TargetComputer, 2)
31       domain_sid = umi2["domain_id"]
32   
33       SubAuthorityCount = domain_sid.GetSubAuthorityCount()
34   
35       # create and init new sid with acct domain Sid + acct Rid
36       sid = pywintypes.SID()
37       sid.Initialize(domain_sid.GetSidIdentifierAuthority(), SubAuthorityCount + 1)
38   
39       # copy existing subauthorities from account domain Sid into
40       # new Sid
41       for i in range(SubAuthorityCount):
42           sid.SetSubAuthority(i, domain_sid.GetSubAuthority(i))
43   
44       # append Rid to new Sid
45       sid.SetSubAuthority(SubAuthorityCount, Rid)
46   
47       name, domain, typ = LookupAccountSid(TargetComputer, sid)
48       return name

50   
51   def main():
52       if len(sys.argv) == 2:
53           targetComputer = sys.argv[1]
54       else:
55           targetComputer = None
56   
57       name = LookupUserGroupFromRid(targetComputer, DOMAIN_USER_RID_ADMIN)
58       print("'Administrator' user name = %s" % (name,))
59   
60       name = LookupAliasFromRid(targetComputer, DOMAIN_ALIAS_RID_ADMINS)
61       print("'Administrators' local group/alias name = %s" % (name,))
62   
63   
64   if __name__ == "__main__":
65       main()
