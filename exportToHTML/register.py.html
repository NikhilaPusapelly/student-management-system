1    """Utilities for registering objects. 
2     
3    This module contains utility functions to register Python objects as 
4    valid COM Servers.  The RegisterServer function provides all information 
5    necessary to allow the COM framework to respond to a request for a COM object, 
6    construct the necessary Python object, and dispatch COM events.

20   def _set_subkeys(keyName, valueDict, base=win32con.HKEY_CLASSES_ROOT):
21       hkey = win32api.RegCreateKey(base, keyName)
22       try:
23           for key, value in valueDict.items():
24               win32api.RegSetValueEx(hkey, key, None, win32con.REG_SZ, value)
25       finally:
26           win32api.RegCloseKey(hkey)

29   def _set_string(path, value, base=win32con.HKEY_CLASSES_ROOT):
30       "Set a string value in the registry."
31   
32       win32api.RegSetValue(base, path, win32con.REG_SZ, value)

35   def _get_string(path, base=win32con.HKEY_CLASSES_ROOT):
36       "Get a string value from the registry."
37   
38       try:
39           return win32api.RegQueryValue(base, path)
40       except win32api.error:
41           return None

45       "Remove a string from the registry."
46   
47       try:
48           win32api.RegDeleteKey(base, path)
49       except win32api.error as xxx_todo_changeme1:
50           (code, fn, msg) = xxx_todo_changeme1.args
51           if code != winerror.ERROR_FILE_NOT_FOUND:
52               raise win32api.error(code, fn, msg)

55   def recurse_delete_key(path, base=win32con.HKEY_CLASSES_ROOT):
56       """Recursively delete registry keys. 
57    
58       This is needed since you can't blast a key when subkeys exist. 
59       """
60       try:
61           h = win32api.RegOpenKey(base, path)
62       except win32api.error as xxx_todo_changeme2:
63           (code, fn, msg) = xxx_todo_changeme2.args
64           if code != winerror.ERROR_FILE_NOT_FOUND:
65               raise win32api.error(code, fn, msg)
66       else:
67           # parent key found and opened successfully. do some work, making sure
68           # to always close the thing (error or no).
69           try:
70               # remove all of the subkeys
71               while 1:
72                   try:
73                       subkeyname = win32api.RegEnumKey(h, 0)
74                   except win32api.error as xxx_todo_changeme:
75                       (code, fn, msg) = xxx_todo_changeme.args
76                       if code != winerror.ERROR_NO_MORE_ITEMS:
77                           raise win32api.error(code, fn, msg)
78                       break
79                   recurse_delete_key(path + "\\" + subkeyname, base)
80   
81               # remove the parent key
82               _remove_key(path, base)
83           finally:
84               win32api.RegCloseKey(h)

96   def _find_localserver_exe(mustfind):
97       if not sys.platform.startswith("win32"):
98           return sys.executable
99       if pythoncom.__file__.find("_d") < 0:
100          exeBaseName = "pythonw.exe"
101      else:
102          exeBaseName = "pythonw_d.exe"
103      # First see if in the same directory as this .EXE
104      exeName = os.path.join(os.path.split(sys.executable)[0], exeBaseName)
105      if not os.path.exists(exeName):
106          # See if in our sys.prefix directory
107          exeName = os.path.join(sys.prefix, exeBaseName)
108      if not os.path.exists(exeName):
109          # See if in our sys.prefix/pcbuild directory (for developers)
110          if "64 bit" in sys.version:
111              exeName = os.path.join(sys.prefix, "PCbuild", "amd64", exeBaseName)
112          else:
113              exeName = os.path.join(sys.prefix, "PCbuild", exeBaseName)
114      if not os.path.exists(exeName):
115          # See if the registry has some info.
116          try:
117              key = "SOFTWARE\\Python\\PythonCore\\%s\\InstallPath" % sys.winver
118              path = win32api.RegQueryValue(win32con.HKEY_LOCAL_MACHINE, key)
119              exeName = os.path.join(path, exeBaseName)
120          except (AttributeError, win32api.error):
121              pass
122      if not os.path.exists(exeName):
123          if mustfind:
124              raise RuntimeError("Can not locate the program '%s'" % exeBaseName)
125          return None
126      return exeName

129  def _find_localserver_module():
130      import win32com.server
131  
132      path = win32com.server.__path__[0]
133      baseName = "localserver"
134      pyfile = os.path.join(path, baseName + ".py")
135      try:
136          os.stat(pyfile)
137      except os.error:
138          # See if we have a compiled extension
139          if __debug__:
140              ext = ".pyc"
141          else:
142              ext = ".pyo"
143          pyfile = os.path.join(path, baseName + ext)
144          try:
145              os.stat(pyfile)
146          except os.error:
147              raise RuntimeError(
148                  "Can not locate the Python module 'win32com.server.%s'" % baseName
149              )
150      return pyfile

153  def RegisterServer(
154      clsid,
155      pythonInstString=None,
156      desc=None,
157      progID=None,
158      verProgID=None,
159      defIcon=None,
160      threadingModel="both",
161      policy=None,
162      catids=[],
163      other={},
164      addPyComCat=None,
165      dispatcher=None,
166      clsctx=None,
167      addnPath=None,
168  ):
169      """Registers a Python object as a COM Server.  This enters almost all necessary 
170      information in the system registry, allowing COM to use the object. 
171   
172      clsid -- The (unique) CLSID of the server. 
173      pythonInstString -- A string holding the instance name that will be created 
174                    whenever COM requests a new object. 
175      desc -- The description of the COM object. 
176      progID -- The user name of this object (eg, Word.Document) 
177      verProgId -- The user name of this version's implementation (eg Word.6.Document) 
178      defIcon -- The default icon for the object. 
179      threadingModel -- The threading model this object supports. 
180      policy -- The policy to use when creating this object. 
181      catids -- A list of category ID's this object belongs in. 
182      other -- A dictionary of extra items to be registered. 
183      addPyComCat -- A flag indicating if the object should be added to the list 
184               of Python servers installed on the machine.  If None (the default) 
185               then it will be registered when running from python source, but 
186               not registered if running in a frozen environment. 
187      dispatcher -- The dispatcher to use when creating this object. 
188      clsctx -- One of the CLSCTX_* constants. 
189      addnPath -- An additional path the COM framework will add to sys.path 
190                  before attempting to create the object. 
191      """
192  
193      ### backwards-compat check
194      ### Certain policies do not require a "class name", just the policy itself.
195      if not pythonInstString and not policy:
196          raise TypeError(
197              "You must specify either the Python Class or Python Policy which implement the COM object."
198          )
199  
200      keyNameRoot = "CLSID\\%s" % str(clsid)
201      _set_string(keyNameRoot, desc)
202  
203      # Also register as an "Application" so DCOM etc all see us.
204      _set_string("AppID\\%s" % clsid, progID)
205      # Depending on contexts requested, register the specified server type.
206      # Set default clsctx.
207      if not clsctx:
208          clsctx = pythoncom.CLSCTX_INPROC_SERVER | pythoncom.CLSCTX_LOCAL_SERVER
209      # And if we are frozen, ignore the ones that don't make sense in this
210      # context.
211      if pythoncom.frozen:
212          assert (
213              sys.frozen
214          ), "pythoncom is frozen, but sys.frozen is not set - don't know the context!"
215          if sys.frozen == "dll":
216              clsctx = clsctx & pythoncom.CLSCTX_INPROC_SERVER
217          else:
218              clsctx = clsctx & pythoncom.CLSCTX_LOCAL_SERVER
219      # Now setup based on the clsctx left over.
220      if clsctx & pythoncom.CLSCTX_INPROC_SERVER:
221          # get the module to use for registration.
222          # nod to Gordon's installer - if sys.frozen and sys.frozendllhandle
223          # exist, then we are being registered via a DLL - use this DLL as the
224          # file name.
225          if pythoncom.frozen:
226              if hasattr(sys, "frozendllhandle"):
227                  dllName = win32api.GetModuleFileName(sys.frozendllhandle)
228              else:
229                  raise RuntimeError(
230                      "We appear to have a frozen DLL, but I don't know the DLL to use"
231                  )
232          else:
233              # Normal case - running from .py file, so register pythoncom's DLL.
234              # Although now we prefer a 'loader' DLL if it exists to avoid some
235              # manifest issues (the 'loader' DLL has a manifest, but pythoncom does not)
236              pythoncom_dir = os.path.dirname(pythoncom.__file__)
237              suffix = "_d" if "_d" in pythoncom.__file__ else ""
238              # Always register with the full path to the DLLs.
239              loadername = os.path.join(
240                  pythoncom_dir,
241                  "pythoncomloader%d%d%s.dll"
242                  % (sys.version_info[0], sys.version_info[1], suffix),
243              )
244              dllName = loadername if os.path.isfile(loadername) else pythoncom.__file__
245  
246          _set_subkeys(
247              keyNameRoot + "\\InprocServer32",
248              {
249                  None: dllName,
250                  "ThreadingModel": threadingModel,
251              },
252          )
253      else:  # Remove any old InProcServer32 registrations
254          _remove_key(keyNameRoot + "\\InprocServer32")
255  
256      if clsctx & pythoncom.CLSCTX_LOCAL_SERVER:
257          if pythoncom.frozen:
258              # If we are frozen, we write "{exe} /Automate", just
259              # like "normal" .EXEs do
260              exeName = win32api.GetShortPathName(sys.executable)
261              command = "%s /Automate" % (exeName,)
262          else:
263              # Running from .py sources - we need to write
264              # 'python.exe win32com\server\localserver.py {clsid}"
265              exeName = _find_localserver_exe(1)
266              exeName = win32api.GetShortPathName(exeName)
267              pyfile = _find_localserver_module()
268              command = '%s "%s" %s' % (exeName, pyfile, str(clsid))
269          _set_string(keyNameRoot + "\\LocalServer32", command)
270      else:  # Remove any old LocalServer32 registrations
271          _remove_key(keyNameRoot + "\\LocalServer32")
272  
273      if pythonInstString:
274          _set_string(keyNameRoot + "\\PythonCOM", pythonInstString)
275      else:
276          _remove_key(keyNameRoot + "\\PythonCOM")
277      if policy:
278          _set_string(keyNameRoot + "\\PythonCOMPolicy", policy)
279      else:
280          _remove_key(keyNameRoot + "\\PythonCOMPolicy")
281  
282      if dispatcher:
283          _set_string(keyNameRoot + "\\PythonCOMDispatcher", dispatcher)
284      else:
285          _remove_key(keyNameRoot + "\\PythonCOMDispatcher")
286  
287      if defIcon:
288          _set_string(keyNameRoot + "\\DefaultIcon", defIcon)
289      else:
290          _remove_key(keyNameRoot + "\\DefaultIcon")
291  
292      if addnPath:
293          _set_string(keyNameRoot + "\\PythonCOMPath", addnPath)
294      else:
295          _remove_key(keyNameRoot + "\\PythonCOMPath")
296  
297      if addPyComCat is None:
298          addPyComCat = pythoncom.frozen == 0
299      if addPyComCat:
300          catids = catids + [CATID_PythonCOMServer]
301  
302      # Set up the implemented categories
303      if catids:
304          regCat = _cat_registrar()
305          regCat.RegisterClassImplCategories(clsid, catids)
306  
307      # set up any other reg values they might have
308      if other:
309          for key, value in other.items():
310              _set_string(keyNameRoot + "\\" + key, value)
311  
312      if progID:
313          # set the progID as the most specific that was given to us
314          if verProgID:
315              _set_string(keyNameRoot + "\\ProgID", verProgID)
316          else:
317              _set_string(keyNameRoot + "\\ProgID", progID)
318  
319          # Set up the root entries - version independent.
320          if desc:
321              _set_string(progID, desc)
322          _set_string(progID + "\\CLSID", str(clsid))
323  
324          # Set up the root entries - version dependent.
325          if verProgID:
326              # point from independent to the current version
327              _set_string(progID + "\\CurVer", verProgID)
328  
329              # point to the version-independent one
330              _set_string(keyNameRoot + "\\VersionIndependentProgID", progID)
331  
332              # set up the versioned progID
333              if desc:
334                  _set_string(verProgID, desc)
335              _set_string(verProgID + "\\CLSID", str(clsid))

338  def GetUnregisterServerKeys(clsid, progID=None, verProgID=None, customKeys=None):
339      """Given a server, return a list of of ("key", root), which are keys recursively 
340      and uncondtionally deleted at unregister or uninstall time. 
341      """
342      # remove the main CLSID registration
343      ret = [("CLSID\\%s" % str(clsid), win32con.HKEY_CLASSES_ROOT)]
344      # remove the versioned ProgID registration
345      if verProgID:
346          ret.append((verProgID, win32con.HKEY_CLASSES_ROOT))
347      # blow away the independent ProgID. we can't leave it since we just
348      # torched the class.
349      ### could potentially check the CLSID... ?
350      if progID:
351          ret.append((progID, win32con.HKEY_CLASSES_ROOT))
352      # The DCOM config tool may write settings to the AppID key for our CLSID
353      ret.append(("AppID\\%s" % str(clsid), win32con.HKEY_CLASSES_ROOT))
354      # Any custom keys?
355      if customKeys:
356          ret = ret + customKeys
357  
358      return ret

362      """Unregisters a Python COM server."""

364      for args in GetUnregisterServerKeys(clsid, progID, verProgID, customKeys):

370      ### another time, though...

377  def GetRegisteredServerOption(clsid, optionName):
378      """Given a CLSID for a server and option name, return the option value"""
379      keyNameRoot = "CLSID\\%s\\%s" % (str(clsid), str(optionName))
380      return _get_string(keyNameRoot)

383  def _get(ob, attr, default=None):
384      try:
385          return getattr(ob, attr)
386      except AttributeError:
387          pass
388      # look down sub-classes
389      try:
390          bases = ob.__bases__
391      except AttributeError:
392          # ob is not a class - no probs.
393          return default
394      for base in bases:
395          val = _get(base, attr, None)
396          if val is not None:
397              return val
398      return default

401  def RegisterClasses(*classes, **flags):
402      quiet = "quiet" in flags and flags["quiet"]
403      debugging = "debug" in flags and flags["debug"]
404      for cls in classes:
405          clsid = cls._reg_clsid_
406          progID = _get(cls, "_reg_progid_")
407          desc = _get(cls, "_reg_desc_", progID)
408          spec = _get(cls, "_reg_class_spec_")
409          verProgID = _get(cls, "_reg_verprogid_")
410          defIcon = _get(cls, "_reg_icon_")
411          threadingModel = _get(cls, "_reg_threading_", "both")
412          catids = _get(cls, "_reg_catids_", [])
413          options = _get(cls, "_reg_options_", {})
414          policySpec = _get(cls, "_reg_policy_spec_")
415          clsctx = _get(cls, "_reg_clsctx_")
416          tlb_filename = _get(cls, "_reg_typelib_filename_")
417          # default to being a COM category only when not frozen.
418          addPyComCat = not _get(cls, "_reg_disable_pycomcat_", pythoncom.frozen != 0)
419          addnPath = None
420          if debugging:
421              # If the class has a debugging dispatcher specified, use it, otherwise
422              # use our default dispatcher.
423              dispatcherSpec = _get(cls, "_reg_debug_dispatcher_spec_")
424              if dispatcherSpec is None:
425                  dispatcherSpec = "win32com.server.dispatcher.DefaultDebugDispatcher"
426              # And remember the debugging flag as servers may wish to use it at runtime.
427              debuggingDesc = "(for debugging)"
428              options["Debugging"] = "1"
429          else:
430              dispatcherSpec = _get(cls, "_reg_dispatcher_spec_")
431              debuggingDesc = ""
432              options["Debugging"] = "0"
433  
434          if spec is None:
435              moduleName = cls.__module__
436              if moduleName == "__main__":
437                  # Use argv[0] to determine the module name.
438                  try:
439                      # Use the win32api to find the case-sensitive name
440                      moduleName = os.path.splitext(
441                          win32api.FindFiles(sys.argv[0])[0][8]
442                      )[0]
443                  except (IndexError, win32api.error):
444                      # Can't find the script file - the user must explicitely set the _reg_... attribute.
445                      raise TypeError(
446                          "Can't locate the script hosting the COM object - please set _reg_class_spec_ in your object"
447                      )
448  
449              spec = moduleName + "." + cls.__name__
450              # Frozen apps don't need their directory on sys.path
451              if not pythoncom.frozen:
452                  scriptDir = os.path.split(sys.argv[0])[0]
453                  if not scriptDir:
454                      scriptDir = "."
455                  addnPath = win32api.GetFullPathName(scriptDir)
456  
457          RegisterServer(
458              clsid,
459              spec,
460              desc,
461              progID,
462              verProgID,
463              defIcon,
464              threadingModel,
465              policySpec,
466              catids,
467              options,
468              addPyComCat,
469              dispatcherSpec,
470              clsctx,
471              addnPath,
472          )
473          if not quiet:
474              print("Registered:", progID or spec, debuggingDesc)
475          # Register the typelibrary
476          if tlb_filename:
477              tlb_filename = os.path.abspath(tlb_filename)
478              typelib = pythoncom.LoadTypeLib(tlb_filename)
479              pythoncom.RegisterTypeLib(typelib, tlb_filename)
480              if not quiet:
481                  print("Registered type library:", tlb_filename)
482      extra = flags.get("finalize_register")
483      if extra:
484          extra()
485

487  def UnregisterClasses(*classes, **flags):
488      quiet = "quiet" in flags and flags["quiet"]
489      for cls in classes:
490          clsid = cls._reg_clsid_
491          progID = _get(cls, "_reg_progid_")
492          verProgID = _get(cls, "_reg_verprogid_")
493          customKeys = _get(cls, "_reg_remove_keys_")
494          unregister_typelib = _get(cls, "_reg_typelib_filename_") is not None
495  
496          UnregisterServer(clsid, progID, verProgID, customKeys)
497          if not quiet:
498              print("Unregistered:", progID or str(clsid))
499          if unregister_typelib:
500              tlb_guid = _get(cls, "_typelib_guid_")
501              if tlb_guid is None:
502                  # I guess I could load the typelib, but they need the GUID anyway.
503                  print("Have typelib filename, but no GUID - can't unregister")
504              else:
505                  major, minor = _get(cls, "_typelib_version_", (1, 0))
506                  lcid = _get(cls, "_typelib_lcid_", 0)
507                  try:
508                      pythoncom.UnRegisterTypeLib(tlb_guid, major, minor, lcid)
509                      if not quiet:
510                          print("Unregistered type library")
511                  except pythoncom.com_error:
512                      pass
513  
514      extra = flags.get("finalize_unregister")
515      if extra:
516          extra()
517

520  # Unregister info is for installers or external uninstallers.
521  # The WISE installer, for example firstly registers the COM server,
522  # then queries for the Unregister info, appending it to its
523  # install log.  Uninstalling the package will the uninstall the server
524  def UnregisterInfoClasses(*classes, **flags):
525      ret = []
526      for cls in classes:
527          clsid = cls._reg_clsid_
528          progID = _get(cls, "_reg_progid_")
529          verProgID = _get(cls, "_reg_verprogid_")
530          customKeys = _get(cls, "_reg_remove_keys_")
531  
532          ret = ret + GetUnregisterServerKeys(clsid, progID, verProgID, customKeys)
533      return ret

536  # Attempt to 're-execute' our current process with elevation.
537  def ReExecuteElevated(flags):
538      import tempfile
539  
540      import win32event  # we've already checked we are running XP above
541      import win32process
542      import winxpgui
543      from win32com.shell import shellcon
544      from win32com.shell.shell import ShellExecuteEx
545  
546      if not flags["quiet"]:
547          print("Requesting elevation and retrying...")
548      new_params = " ".join(['"' + a + '"' for a in sys.argv])
549      # If we aren't already in unattended mode, we want our sub-process to
550      # be.
551      if not flags["unattended"]:
552          new_params += " --unattended"
553      # specifying the parent means the dialog is centered over our window,
554      # which is a good usability clue.
555      # hwnd is unlikely on the command-line, but flags may come from elsewhere
556      hwnd = flags.get("hwnd", None)
557      if hwnd is None:
558          try:
559              hwnd = winxpgui.GetConsoleWindow()
560          except winxpgui.error:
561              hwnd = 0
562      # Redirect output so we give the user some clue what went wrong.  This
563      # also means we need to use COMSPEC.  However, the "current directory"
564      # appears to end up ignored - so we execute things via a temp batch file.
565      tempbase = tempfile.mktemp("pycomserverreg")
566      outfile = tempbase + ".out"
567      batfile = tempbase + ".bat"
568  
569      # If registering from pythonwin, need to run python console instead since
570      #  pythonwin will just open script for editting
571      current_exe = os.path.split(sys.executable)[1].lower()
572      exe_to_run = None
573      if current_exe == "pythonwin.exe":
574          exe_to_run = os.path.join(sys.prefix, "python.exe")
575      elif current_exe == "pythonwin_d.exe":
576          exe_to_run = os.path.join(sys.prefix, "python_d.exe")
577      if not exe_to_run or not os.path.exists(exe_to_run):
578          exe_to_run = sys.executable
579  
580      try:
581          batf = open(batfile, "w")
582          try:
583              cwd = os.getcwd()
584              print("@echo off", file=batf)
585              # nothing is 'inherited' by the elevated process, including the
586              # environment.  I wonder if we need to set more?
587              print("set PYTHONPATH=%s" % os.environ.get("PYTHONPATH", ""), file=batf)
588              # may be on a different drive - select that before attempting to CD.
589              print(os.path.splitdrive(cwd)[0], file=batf)
590              print('cd "%s"' % os.getcwd(), file=batf)
591              print(
592                  '%s %s > "%s" 2>&1'
593                  % (win32api.GetShortPathName(exe_to_run), new_params, outfile),
594                  file=batf,
595              )
596          finally:
597              batf.close()
598          executable = os.environ.get("COMSPEC", "cmd.exe")
599          rc = ShellExecuteEx(
600              hwnd=hwnd,
601              fMask=shellcon.SEE_MASK_NOCLOSEPROCESS,
602              lpVerb="runas",
603              lpFile=executable,
604              lpParameters='/C "%s"' % batfile,
605              nShow=win32con.SW_SHOW,
606          )
607          hproc = rc["hProcess"]
608          win32event.WaitForSingleObject(hproc, win32event.INFINITE)
609          exit_code = win32process.GetExitCodeProcess(hproc)
610          outf = open(outfile)
611          try:
612              output = outf.read()
613          finally:
614              outf.close()
615  
616          if exit_code:
617              # Even if quiet you get to see this message.
618              print("Error: registration failed (exit code %s)." % exit_code)
619          # if we are quiet then the output if likely to already be nearly
620          # empty, so always print it.
621          print(output, end=" ")
622      finally:
623          for f in (outfile, batfile):
624              try:
625                  os.unlink(f)
626              except os.error as exc:
627                  print("Failed to remove tempfile '%s': %s" % (f, exc))

630  def UseCommandLine(*classes, **flags):
631      unregisterInfo = "--unregister_info" in sys.argv
632      unregister = "--unregister" in sys.argv
633      flags["quiet"] = flags.get("quiet", 0) or "--quiet" in sys.argv
634      flags["debug"] = flags.get("debug", 0) or "--debug" in sys.argv
635      flags["unattended"] = flags.get("unattended", 0) or "--unattended" in sys.argv
636      if unregisterInfo:
637          return UnregisterInfoClasses(*classes, **flags)
638      try:
639          if unregister:
640              UnregisterClasses(*classes, **flags)
641          else:
642              RegisterClasses(*classes, **flags)
643      except win32api.error as exc:
644          # If we are on xp+ and have "access denied", retry using
645          # ShellExecuteEx with 'runas' verb to force elevation (vista) and/or
646          # admin login dialog (vista/xp)
647          if (
648              flags["unattended"]
649              or exc.winerror != winerror.ERROR_ACCESS_DENIED
650              or sys.getwindowsversion()[0] < 5
651          ):
652              raise
653          ReExecuteElevated(flags)
