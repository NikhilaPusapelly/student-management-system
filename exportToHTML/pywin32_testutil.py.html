15   # The test suite has lots of string constants containing binary data, but
16   # the strings are used in various "bytes" contexts.
17   def str2bytes(sval):
18       if sys.version_info < (3, 0) and isinstance(sval, str):
19           sval = sval.decode("latin1")
20       return sval.encode("latin1")
21

23   # Sometimes we want to pass a string that should explicitly be treated as
24   # a memory blob.
25   def str2memory(sval):
26       if sys.version_info < (3, 0):
27           return buffer(sval)
28       # py3k.
29       return memoryview(sval.encode("latin1"))
30   
31   
32   # Sometimes we want to pass an object that exposes its memory
33   def ob2memory(ob):
34       if sys.version_info < (3, 0):
35           return buffer(ob)
36       # py3k.
37       return memoryview(ob)

45   # This is a specialized TestCase adaptor which wraps a real test.
46   class LeakTestCase(unittest.TestCase):
47       """An 'adaptor' which takes another test.  In debug builds we execute the 
48       test once to remove one-off side-effects, then capture the total 
49       reference count, then execute the test a few times.  If the total 
50       refcount at the end is greater than we first captured, we have a leak! 
51    
52       In release builds the test is executed just once, as normal. 
53    
54       Generally used automatically by the test runner - you can safely 
55       ignore this. 
56       """
57   
58       def __init__(self, real_test):
59           unittest.TestCase.__init__(self)
60           self.real_test = real_test
61           self.num_test_cases = 1
62           self.num_leak_iters = 2  # seems to be enough!
63           if hasattr(sys, "gettotalrefcount"):
64               self.num_test_cases = self.num_test_cases + self.num_leak_iters
65   
66       def countTestCases(self):
67           return self.num_test_cases
68   
69       def __call__(self, result=None):
70           # For the COM suite's sake, always ensure we don't leak
71           # gateways/interfaces
72           from pythoncom import _GetGatewayCount, _GetInterfaceCount
73   
74           gc.collect()
75           ni = _GetInterfaceCount()
76           ng = _GetGatewayCount()
77           self.real_test(result)
78           # Failed - no point checking anything else
79           if result.shouldStop or not result.wasSuccessful():
80               return
81           self._do_leak_tests(result)
82           gc.collect()
83           lost_i = _GetInterfaceCount() - ni
84           lost_g = _GetGatewayCount() - ng
85           if lost_i or lost_g:
86               msg = "%d interface objects and %d gateway objects leaked" % (
87                   lost_i,
88                   lost_g,
89               )
90               exc = AssertionError(msg)
91               result.addFailure(self.real_test, (exc.__class__, exc, None))
92   
93       def runTest(self):
94           assert 0, "not used"
95   
96       def _do_leak_tests(self, result=None):
97           try:
98               gtrc = sys.gettotalrefcount
99           except AttributeError:
100              return  # can't do leak tests in this build
101          # Assume already called once, to prime any caches etc
102          gc.collect()
103          trc = gtrc()
104          for i in range(self.num_leak_iters):
105              self.real_test(result)
106              if result.shouldStop:
107                  break
108          del i  # created after we remembered the refcount!
109          # int division here means one or 2 stray references won't force
110          # failure, but one per loop
111          gc.collect()
112          lost = (gtrc() - trc) // self.num_leak_iters
113          if lost < 0:
114              msg = "LeakTest: %s appeared to gain %d references!!" % (
115                  self.real_test,
116                  -lost,
117              )
118              result.addFailure(self.real_test, (AssertionError, msg, None))
119          if lost > 0:
120              msg = "LeakTest: %s lost %d references" % (self.real_test, lost)
121              exc = AssertionError(msg)
122              result.addFailure(self.real_test, (exc.__class__, exc, None))

125  class TestLoader(unittest.TestLoader):
126      def loadTestsFromTestCase(self, testCaseClass):
127          """Return a suite of all tests cases contained in testCaseClass"""
128          leak_tests = []
129          for name in self.getTestCaseNames(testCaseClass):
130              real_test = testCaseClass(name)
131              leak_test = self._getTestWrapper(real_test)
132              leak_tests.append(leak_test)
133          return self.suiteClass(leak_tests)
134  
135      def fixupTestsForLeakTests(self, test):
136          if isinstance(test, unittest.TestSuite):
137              test._tests = [self.fixupTestsForLeakTests(t) for t in test._tests]
138              return test
139          else:
140              # just a normal test case.
141              return self._getTestWrapper(test)
142  
143      def _getTestWrapper(self, test):
144          # one or 2 tests in the COM test suite set this...
145          no_leak_tests = getattr(test, "no_leak_tests", False)
146          if no_leak_tests:
147              print("Test says it doesn't want leak tests!")
148              return test
149          return LeakTestCase(test)
150  
151      def loadTestsFromModule(self, mod):
152          if hasattr(mod, "suite"):
153              tests = mod.suite()
154          else:
155              tests = unittest.TestLoader.loadTestsFromModule(self, mod)
156          return self.fixupTestsForLeakTests(tests)
157  
158      def loadTestsFromName(self, name, module=None):
159          test = unittest.TestLoader.loadTestsFromName(self, name, module)
160          if isinstance(test, unittest.TestSuite):
161              pass  # hmmm? print "Don't wrap suites yet!", test._tests
162          elif isinstance(test, unittest.TestCase):
163              test = self._getTestWrapper(test)
164          else:
165              print("XXX - what is", test)
166          return test

185  def check_is_admin():
186      global _is_admin
187      if _is_admin is None:
188          import pythoncom
189          from win32com.shell.shell import IsUserAnAdmin
190  
191          try:
192              _is_admin = IsUserAnAdmin()
193          except pythoncom.com_error as exc:
194              if exc.hresult != winerror.E_NOTIMPL:
195                  raise
196              # not impl on this platform - must be old - assume is admin
197              _is_admin = True
198      return _is_admin

201  # Find a test "fixture" (eg, binary test file) expected to be very close to
202  # the test being run.
203  # If the tests are being run from the "installed" version, then these fixtures
204  # probably don't exist - the test is "skipped".
205  # But it's fatal if we think we might be running from a pywin32 source tree.
206  def find_test_fixture(basename, extra_dir="."):
207      # look for the test file in various places
208      candidates = [
209          os.path.dirname(sys.argv[0]),
210          extra_dir,
211          ".",
212      ]
213      for candidate in candidates:
214          fname = os.path.join(candidate, basename)
215          if os.path.isfile(fname):
216              return fname
217      else:
218          # Can't find it - see if this is expected or not.
219          # This module is typically always in the installed dir, so use argv[0]
220          this_file = os.path.normcase(os.path.abspath(sys.argv[0]))
221          dirs_to_check = site.getsitepackages()[:]
222          if site.USER_SITE:
223              dirs_to_check.append(site.USER_SITE)
224  
225          for d in dirs_to_check:
226              d = os.path.normcase(d)
227              if os.path.commonprefix([this_file, d]) == d:
228                  # looks like we are in an installed Python, so skip the text.
229                  raise TestSkipped(f"Can't find test fixture '{fname}'")
230          # Looks like we are running from source, so this is fatal.
231          raise RuntimeError(f"Can't find test fixture '{fname}'")

246  # The 'TestResult' subclass that records the failures and has the special
247  # handling for the TestSkipped exception.
248  class TestResult(TextTestResult):
249      def __init__(self, *args, **kw):
250          super(TestResult, self).__init__(*args, **kw)
251          self.skips = {}  # count of skips for each reason.
252  
253      def addError(self, test, err):
254          """Called when an error has occurred. 'err' is a tuple of values as 
255          returned by sys.exc_info(). 
256          """
257          # translate a couple of 'well-known' exceptions into 'skipped'
258          import pywintypes
259  
260          exc_val = err[1]
261          # translate ERROR_ACCESS_DENIED for non-admin users to be skipped.
262          # (access denied errors for an admin user aren't expected.)
263          if (
264              isinstance(exc_val, pywintypes.error)
265              and exc_val.winerror in non_admin_error_codes
266              and not check_is_admin()
267          ):
268              exc_val = TestSkipped(exc_val)
269          # and COM errors due to objects not being registered (the com test
270          # suite will attempt to catch this and handle it itself if the user
271          # is admin)
272          elif isinstance(exc_val, pywintypes.com_error) and exc_val.hresult in [
273              winerror.CO_E_CLASSSTRING,
274              winerror.REGDB_E_CLASSNOTREG,
275              winerror.TYPE_E_LIBNOTREGISTERED,
276          ]:
277              exc_val = TestSkipped(exc_val)
278          # NotImplemented generally means the platform doesn't support the
279          # functionality.
280          elif isinstance(exc_val, NotImplementedError):
281              exc_val = TestSkipped(NotImplementedError)
282  
283          if isinstance(exc_val, TestSkipped):
284              reason = exc_val.args[0]
285              # if the reason itself is another exception, get its args.
286              try:
287                  reason = tuple(reason.args)
288              except (AttributeError, TypeError):
289                  pass
290              self.skips.setdefault(reason, 0)
291              self.skips[reason] += 1
292              if self.showAll:
293                  self.stream.writeln("SKIP (%s)" % (reason,))
294              elif self.dots:
295                  self.stream.write("S")
296                  self.stream.flush()
297              return
298          super(TestResult, self).addError(test, err)
299  
300      def printErrors(self):
301          super(TestResult, self).printErrors()
302          for reason, num_skipped in self.skips.items():
303              self.stream.writeln("SKIPPED: %d tests - %s" % (num_skipped, reason))

321  # A convenient entry-point - if used, 'SKIPPED' exceptions will be supressed.
322  def testmain(*args, **kw):
323      new_kw = kw.copy()
324      if "testLoader" not in new_kw:
325          new_kw["testLoader"] = TestLoader()
326      program_class = new_kw.get("testProgram", TestProgram)
327      program_class(*args, **new_kw)
