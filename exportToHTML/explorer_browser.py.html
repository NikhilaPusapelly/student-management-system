5    import sys
6    
7    import pythoncom
8    import win32api
9    import win32con
10   import win32gui
11   from win32com.server.util import unwrap, wrap
12   from win32com.shell import shell, shellcon

14   # event handler for the browser.
15   IExplorerBrowserEvents_Methods = """OnNavigationComplete OnNavigationFailed  
16                                       OnNavigationPending OnViewCreated""".split()
17   
18   
19   class EventHandler:
20       _com_interfaces_ = [shell.IID_IExplorerBrowserEvents]
21       _public_methods_ = IExplorerBrowserEvents_Methods
22   
23       def OnNavigationComplete(self, pidl):
24           print("OnNavComplete", pidl)
25   
26       def OnNavigationFailed(self, pidl):
27           print("OnNavigationFailed", pidl)
28   
29       def OnNavigationPending(self, pidl):
30           print("OnNavigationPending", pidl)
31   
32       def OnViewCreated(self, view):
33           print("OnViewCreated", view)
34           # And if our demo view has been registered, it may well
35           # be that view!
36           try:
37               pyview = unwrap(view)
38               print("and look - its a Python implemented view!", pyview)
39           except ValueError:
40               pass

43   class MainWindow:
44       def __init__(self):
45           message_map = {
46               win32con.WM_DESTROY: self.OnDestroy,
47               win32con.WM_COMMAND: self.OnCommand,
48               win32con.WM_SIZE: self.OnSize,
49           }
50           # Register the Window class.
51           wc = win32gui.WNDCLASS()
52           hinst = wc.hInstance = win32api.GetModuleHandle(None)
53           wc.lpszClassName = "test_explorer_browser"
54           wc.lpfnWndProc = message_map  # could also specify a wndproc.
55           classAtom = win32gui.RegisterClass(wc)
56           # Create the Window.
57           style = win32con.WS_OVERLAPPEDWINDOW | win32con.WS_VISIBLE
58           self.hwnd = win32gui.CreateWindow(
59               classAtom,
60               "Python IExplorerBrowser demo",
61               style,
62               0,
63               0,
64               win32con.CW_USEDEFAULT,
65               win32con.CW_USEDEFAULT,
66               0,
67               0,
68               hinst,
69               None,
70           )
71           eb = pythoncom.CoCreateInstance(
72               shellcon.CLSID_ExplorerBrowser,
73               None,
74               pythoncom.CLSCTX_ALL,
75               shell.IID_IExplorerBrowser,
76           )
77           # as per MSDN docs, hook up events early
78           self.event_cookie = eb.Advise(wrap(EventHandler()))
79   
80           eb.SetOptions(shellcon.EBO_SHOWFRAMES)
81           rect = win32gui.GetClientRect(self.hwnd)
82           # Set the flags such that the folders autoarrange and non web view is presented
83           flags = (shellcon.FVM_LIST, shellcon.FWF_AUTOARRANGE | shellcon.FWF_NOWEBVIEW)
84           eb.Initialize(self.hwnd, rect, (0, shellcon.FVM_DETAILS))
85           if len(sys.argv) == 2:
86               # If an arg was specified, ask the desktop parse it.
87               # You can pass anything explorer accepts as its '/e' argument -
88               # eg, "::{guid}\::{guid}" etc.
89               # "::{20D04FE0-3AEA-1069-A2D8-08002B30309D}" is "My Computer"
90               pidl = shell.SHGetDesktopFolder().ParseDisplayName(0, None, sys.argv[1])[1]
91           else:
92               # And start browsing at the root of the namespace.
93               pidl = []
94           eb.BrowseToIDList(pidl, shellcon.SBSP_ABSOLUTE)
95           # and for some reason the "Folder" view in the navigator pane doesn't
96           # magically synchronize itself - so let's do that ourself.
97           # Get the tree control.
98           sp = eb.QueryInterface(pythoncom.IID_IServiceProvider)
99           try:
100              tree = sp.QueryService(
101                  shell.IID_INameSpaceTreeControl, shell.IID_INameSpaceTreeControl
102              )
103          except pythoncom.com_error as exc:
104              # this should really only fail if no "nav" frame exists...
105              print(
106                  "Strange - failed to get the tree control even though "
107                  "we asked for a EBO_SHOWFRAMES"
108              )
109              print(exc)
110          else:
111              # get the IShellItem for the selection.
112              si = shell.SHCreateItemFromIDList(pidl, shell.IID_IShellItem)
113              # set it to selected.
114              tree.SetItemState(si, shellcon.NSTCIS_SELECTED, shellcon.NSTCIS_SELECTED)
115  
116          # eb.FillFromObject(None, shellcon.EBF_NODROPTARGET);
117          # eb.SetEmptyText("No known folders yet...");
118          self.eb = eb
119  
120      def OnCommand(self, hwnd, msg, wparam, lparam):
121          pass
122  
123      def OnDestroy(self, hwnd, msg, wparam, lparam):
124          print("tearing down ExplorerBrowser...")
125          self.eb.Unadvise(self.event_cookie)
126          self.eb.Destroy()
127          self.eb = None
128          print("shutting down app...")
129          win32gui.PostQuitMessage(0)
130  
131      def OnSize(self, hwnd, msg, wparam, lparam):
132          x = win32api.LOWORD(lparam)
133          y = win32api.HIWORD(lparam)
134          self.eb.SetRect(None, (0, 0, x, y))
