viewstate.py 
1    """ 
2    Demonstrates how to propagate a folder's view state to all its subfolders 
3    The format of the ColInfo stream is apparently undocumented, but 
4    it can be read raw from one folder and copied to another's view state. 
5    """
6    
7    import os
8    import sys
9    
10   import pythoncom
11   from win32com.shell import shell, shellcon
12   
13   template_folder = os.path.split(sys.executable)[0]
14   print("Template folder:", template_folder)
15   template_pidl = shell.SHILCreateFromPath(template_folder, 0)[0]
16   template_pb = shell.SHGetViewStatePropertyBag(
17       template_pidl,
18       "Shell",
19       shellcon.SHGVSPB_FOLDERNODEFAULTS,
20       pythoncom.IID_IPropertyBag,
21   )
22   
23   # Column info has to be read as a stream
24   # This may blow up if folder has never been opened in Explorer and has no ColInfo yet
25   template_iunk = template_pb.Read("ColInfo", pythoncom.VT_UNKNOWN)
26   template_stream = template_iunk.QueryInterface(pythoncom.IID_IStream)
27   streamsize = template_stream.Stat()[2]
28   template_colinfo = template_stream.Read(streamsize)
29   
30   
31   def update_colinfo(not_used, dir_name, fnames):
32       for fname in fnames:
33           full_fname = os.path.join(dir_name, fname)
34           if os.path.isdir(full_fname):
35               print(full_fname)
36               pidl = shell.SHILCreateFromPath(full_fname, 0)[0]
37               pb = shell.SHGetViewStatePropertyBag(
38                   pidl,
39                   "Shell",
40                   shellcon.SHGVSPB_FOLDERNODEFAULTS,
41                   pythoncom.IID_IPropertyBag,
42               )
43               ## not all folders already have column info, and we're replacing it anyway
44               pb.Write("ColInfo", template_stream)
45               iunk = pb.Read("ColInfo", pythoncom.VT_UNKNOWN)
46               s = iunk.QueryInterface(pythoncom.IID_IStream)
47               s.Write(template_colinfo)
48               s = None
49               ## attribute names read from registry, can't find any way to enumerate IPropertyBag
50               for attr in (
51                   "Address",
52                   "Buttons",
53                   "Col",
54                   "Vid",
55                   "WFlags",
56                   "FFlags",
57                   "Sort",
58                   "SortDir",
59                   "ShowCmd",
60                   "FolderType",
61                   "Mode",
62                   "Rev",
63               ):
64                   pb.Write(attr, template_pb.Read(attr))
65               pb = None
66   
67   
68   os.path.walk(template_folder, update_colinfo, None)
