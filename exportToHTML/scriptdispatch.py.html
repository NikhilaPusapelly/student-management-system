23   def _is_callable(obj):
24       return type(obj) in [types.FunctionType, types.MethodType]
25       # ignore hasattr(obj, "__call__") as this means all COM objects!
26   
27   
28   class ScriptDispatch:
29       _public_methods_ = []
30   
31       def __init__(self, engine, scriptNamespace):
32           self.engine = engine
33           self.scriptNamespace = scriptNamespace
34   
35       def _dynamic_(self, name, lcid, wFlags, args):
36           # Ensure any newly added items are available.
37           self.engine.RegisterNewNamedItems()
38           self.engine.ProcessNewNamedItemsConnections()
39           if wFlags & pythoncom.INVOKE_FUNC:
40               # attempt to call a function
41               try:
42                   func = getattr(self.scriptNamespace, name)
43                   if not _is_callable(func):
44                       raise AttributeError(name)  # Not a function.
45                   realArgs = []
46                   for arg in args:
47                       if type(arg) == PyIDispatchType:
48                           realArgs.append(Dispatch(arg))
49                       else:
50                           realArgs.append(arg)
51                   # xxx - todo - work out what code block to pass???
52                   return self.engine.ApplyInScriptedSection(None, func, tuple(realArgs))
53   
54               except AttributeError:
55                   if not wFlags & pythoncom.DISPATCH_PROPERTYGET:
56                       raise COMException(scode=winerror.DISP_E_MEMBERNOTFOUND)
57           if wFlags & pythoncom.DISPATCH_PROPERTYGET:
58               # attempt to get a property
59               try:
60                   ret = getattr(self.scriptNamespace, name)
61                   if _is_callable(ret):
62                       raise AttributeError(name)  # Not a property.
63               except AttributeError:
64                   raise COMException(scode=winerror.DISP_E_MEMBERNOTFOUND)
65               except COMException as instance:
66                   raise
67               except:
68                   ret = self.engine.HandleException()
69               return ret
70   
71           raise COMException(scode=winerror.DISP_E_MEMBERNOTFOUND)

74   class StrictDynamicPolicy(win32com.server.policy.DynamicPolicy):
75       def _wrap_(self, object):
76           win32com.server.policy.DynamicPolicy._wrap_(self, object)
77           if hasattr(self._obj_, "scriptNamespace"):
78               for name in dir(self._obj_.scriptNamespace):
79                   self._dyn_dispid_to_name_[self._getdispid_(name, 0)] = name
80   
81       def _getmembername_(self, dispid):
82           try:
83               return str(self._dyn_dispid_to_name_[dispid])
84           except KeyError:
85               raise COMException(scode=winerror.DISP_E_UNKNOWNNAME, desc="Name not found")
86   
87       def _getdispid_(self, name, fdex):
88           try:
89               func = getattr(self._obj_.scriptNamespace, str(name))
90           except AttributeError:
91               raise COMException(scode=winerror.DISP_E_MEMBERNOTFOUND)
92           #       if not _is_callable(func):
93           return win32com.server.policy.DynamicPolicy._getdispid_(self, name, fdex)
94

96   def _wrap_debug(obj):
97       return win32com.server.util.wrap(
98           obj,
99           usePolicy=StrictDynamicPolicy,
100          useDispatcher=win32com.server.policy.DispatcherWin32trace,
101      )

104  def _wrap_nodebug(obj):
105      return win32com.server.util.wrap(obj, usePolicy=StrictDynamicPolicy)

109      _wrap = _wrap_debug
