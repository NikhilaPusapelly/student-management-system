19       def load(self, filename):
20           # Get an IPersist interface
21           # which allows save/restore of object to/from files
22           self._base.QueryInterface(pythoncom.IID_IPersistFile).Load(filename)
23   
24       def save(self, filename):
25           self._base.QueryInterface(pythoncom.IID_IPersistFile).Save(filename, 0)
26   
27       def __getattr__(self, name):
28           if name != "_base":
29               return getattr(self._base, name)
