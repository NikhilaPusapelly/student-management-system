52   class Dib:
53       """ 
54       A Windows bitmap with the given mode and size.  The mode can be one of "1", 
55       "L", "P", or "RGB". 
56    
57       If the display requires a palette, this constructor creates a suitable 
58       palette and associates it with the image. For an "L" image, 128 graylevels 
59       are allocated. For an "RGB" image, a 6x6x6 colour cube is used, together 
60       with 20 graylevels. 
61    
62       To make sure that palettes work properly under Windows, you must call the 
63       ``palette`` method upon certain events from Windows. 
64    
65       :param image: Either a PIL image, or a mode string. If a mode string is 
66                     used, a size must also be given.  The mode can be one of "1", 
67                     "L", "P", or "RGB". 
68       :param size: If the first argument is a mode string, this 
69                    defines the size of the image. 
70       """
71   
72       def __init__(self, image, size=None):
73           if hasattr(image, "mode") and hasattr(image, "size"):
74               mode = image.mode
75               size = image.size
76           else:
77               mode = image
78               image = None
79           if mode not in ["1", "L", "P", "RGB"]:
80               mode = Image.getmodebase(mode)
81           self.image = Image.core.display(mode, size)
82           self.mode = mode
83           self.size = size
84           if image:
85               self.paste(image)
86   
87       def expose(self, handle):
88           """ 
89           Copy the bitmap contents to a device context. 
90    
91           :param handle: Device context (HDC), cast to a Python integer, or an 
92                          HDC or HWND instance.  In PythonWin, you can use 
93                          ``CDC.GetHandleAttrib()`` to get a suitable handle. 
94           """
95           if isinstance(handle, HWND):
96               dc = self.image.getdc(handle)
97               try:
98                   result = self.image.expose(dc)
99               finally:
100                  self.image.releasedc(handle, dc)
101          else:
102              result = self.image.expose(handle)
103          return result
104  
105      def draw(self, handle, dst, src=None):
106          """ 
107          Same as expose, but allows you to specify where to draw the image, and 
108          what part of it to draw. 
109   
110          The destination and source areas are given as 4-tuple rectangles. If 
111          the source is omitted, the entire image is copied. If the source and 
112          the destination have different sizes, the image is resized as 
113          necessary. 
114          """
115          if not src:
116              src = (0, 0) + self.size
117          if isinstance(handle, HWND):
118              dc = self.image.getdc(handle)
119              try:
120                  result = self.image.draw(dc, dst, src)
121              finally:
122                  self.image.releasedc(handle, dc)
123          else:
124              result = self.image.draw(handle, dst, src)
125          return result
126  
127      def query_palette(self, handle):
128          """ 
129          Installs the palette associated with the image in the given device 
130          context. 
131   
132          This method should be called upon **QUERYNEWPALETTE** and 
133          **PALETTECHANGED** events from Windows. If this method returns a 
134          non-zero value, one or more display palette entries were changed, and 
135          the image should be redrawn. 
136   
137          :param handle: Device context (HDC), cast to a Python integer, or an 
138                         HDC or HWND instance. 
139          :return: A true value if one or more entries were changed (this 
140                   indicates that the image should be redrawn). 
141          """
142          if isinstance(handle, HWND):
143              handle = self.image.getdc(handle)
144              try:
145                  result = self.image.query_palette(handle)
146              finally:
147                  self.image.releasedc(handle, handle)
148          else:
149              result = self.image.query_palette(handle)
150          return result
151  
152      def paste(self, im, box=None):
153          """ 
154          Paste a PIL image into the bitmap image. 
155   
156          :param im: A PIL image.  The size must match the target region. 
157                     If the mode does not match, the image is converted to the 
158                     mode of the bitmap image. 
159          :param box: A 4-tuple defining the left, upper, right, and 
160                      lower pixel coordinate.  See :ref:`coordinate-system`. If 
161                      None is given instead of a tuple, all of the image is 
162                      assumed. 
163          """
164          im.load()
165          if self.mode != im.mode:
166              im = im.convert(self.mode)
167          if box:
168              self.image.paste(im.im, box)
169          else:
170              self.image.paste(im.im)
171  
172      def frombytes(self, buffer):
173          """ 
174          Load display memory contents from byte data. 
175   
176          :param buffer: A buffer containing display data (usually 
177                         data returned from :py:func:`~PIL.ImageWin.Dib.tobytes`) 
178          """
179          return self.image.frombytes(buffer)
180  
181      def tobytes(self):
182          """ 
183          Copy display memory contents to bytes object. 
184   
185          :return: A bytes object containing display data. 
186          """
187          return self.image.tobytes()

190  class Window:
191      """Create a Window with the given title size."""
192  
193      def __init__(self, title="PIL", width=None, height=None):
194          self.hwnd = Image.core.createwindow(
195              title, self.__dispatcher, width or 0, height or 0
196          )
197  
198      def __dispatcher(self, action, *args):
199          return getattr(self, "ui_handle_" + action)(*args)
200  
201      def ui_handle_clear(self, dc, x0, y0, x1, y1):
202          pass
203  
204      def ui_handle_damage(self, x0, y0, x1, y1):
205          pass
206  
207      def ui_handle_destroy(self):
208          pass
209  
210      def ui_handle_repair(self, dc, x0, y0, x1, y1):
211          pass
212  
213      def ui_handle_resize(self, width, height):
214          pass
215  
216      def mainloop(self):
217          Image.core.eventloop()
218

220  class ImageWindow(Window):
221      """Create an image window which displays the given image."""
222  
223      def __init__(self, image, title="PIL"):
224          if not isinstance(image, Dib):
225              image = Dib(image)
226          self.image = image
227          width, height = image.size
228          super().__init__(title, width=width, height=height)
229  
230      def ui_handle_repair(self, dc, x0, y0, x1, y1):
231          self.image.draw(dc, (x0, y0, x1, y1))
