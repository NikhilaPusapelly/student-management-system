50   def getsubdirs(d):
51       dlist = []
52       flist = glob.glob(d + "\\*")
53       for f in flist:
54           if os.path.isdir(f):
55               dlist.append(f)
56               dlist = dlist + getsubdirs(f)
57       return dlist

60   class dirpath:
61       def __init__(self, str, recurse=0):
62           dp = str.split(";")
63           dirs = {}
64           for d in dp:
65               if os.path.isdir(d):
66                   d = d.lower()
67                   if d not in dirs:
68                       dirs[d] = None
69                       if recurse:
70                           subdirs = getsubdirs(d)
71                           for sd in subdirs:
72                               sd = sd.lower()
73                               if sd not in dirs:
74                                   dirs[sd] = None
75               elif os.path.isfile(d):
76                   pass
77               else:
78                   x = None
79                   if d in os.environ:
80                       x = dirpath(os.environ[d])
81                   elif d[:5] == "HKEY_":
82                       keystr = d.split("\\")
83                       try:
84                           root = eval("win32con." + keystr[0])
85                       except:
86                           win32ui.MessageBox(
87                               "Can't interpret registry key name '%s'" % keystr[0]
88                           )
89                       try:
90                           subkey = "\\".join(keystr[1:])
91                           val = win32api.RegQueryValue(root, subkey)
92                           if val:
93                               x = dirpath(val)
94                           else:
95                               win32ui.MessageBox(
96                                   "Registry path '%s' did not return a path entry" % d
97                               )
98                       except:
99                           win32ui.MessageBox(
100                              "Can't interpret registry key value: %s" % keystr[1:]
101                          )
102                  else:
103                      win32ui.MessageBox("Directory '%s' not found" % d)
104                  if x:
105                      for xd in x:
106                          if xd not in dirs:
107                              dirs[xd] = None
108                              if recurse:
109                                  subdirs = getsubdirs(xd)
110                                  for sd in subdirs:
111                                      sd = sd.lower()
112                                      if sd not in dirs:
113                                          dirs[sd] = None
114          self.dirs = []
115          for d in dirs.keys():
116              self.dirs.append(d)

515      def OnAddComment(self, cmd, code):
516          addspecific = cmd == ID_ADDPYCHECKNO2
517          _ = list(self.GetSel())
518          _.sort()
519          start, end = _
520          line_start, line_end = self.LineFromChar(start), self.LineFromChar(end)
521          first = 1
522          for i in range(line_start, line_end + 1):
523              line = self.GetLine(i)
524              m = regexGrep.match(line)
525              if m:
526                  if first:
527                      first = 0
528                      cmnt = dialog.GetSimpleInput(
529                          "Add to %s lines" % (line_end - line_start + 1),
530                          addspecific
531                          and "  #$pycheck_no=%(errtext)s"
532                          or "  #$pycheck_no",
533                      )
534                      if not cmnt:
535                          return 0
536                  ##import pywin.debugger;pywin.debugger.set_trace()
537                  fname = m.group(1)
538                  line = int(m.group(2))
539                  view = scriptutils.JumpToDocument(fname, line)
540                  pos = view.LineIndex(line) - 1
541                  if view.GetTextRange(pos - 1, pos) in ("\r", "\n"):
542                      pos -= 1
543                  view.SetSel(pos, pos)
544                  errtext = m.group(3)
545                  if start != end and line_start == line_end:
546                      errtext = self.GetSelText()
547                  errtext = repr(re.escape(errtext).replace("\ ", " "))
548                  view.ReplaceSel(addspecific and cmnt % locals() or cmnt)
549          return 0

698      def getMore(self, section, key):
699          self.UpdateData(1)
700          # get the items out of the ini file
701          ini = win32ui.GetProfileFileName()
702          secitems = win32api.GetProfileSection(section, ini)
703          items = []
704          for secitem in secitems:
705              items.append(secitem.split("=")[1])
706          dlg = TheParamsDialog(items)
707          if dlg.DoModal() == win32con.IDOK:
708              itemstr = ";".join(dlg.getItems())
709              self._obj_.data[key] = itemstr
710              # update the ini file with dlg.getNew()
711              i = 0
712              newitems = dlg.getNew()
713              if newitems:
714                  items = items + newitems
715                  for item in items:
716                      win32api.WriteProfileVal(section, repr(i), item, ini)
717                      i = i + 1
718              self.UpdateData(0)

728                  self.GetDlgItem(id).SetFocus()
