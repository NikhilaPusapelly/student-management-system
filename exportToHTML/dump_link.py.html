3    import glob
4    import os
5    import sys
6    
7    import pythoncom
8    from win32com.shell import shell, shellcon
9    from win32com.storagecon import *

11   
12   def DumpLink(fname):
13       shellLink = pythoncom.CoCreateInstance(
14           shell.CLSID_ShellLink,
15           None,
16           pythoncom.CLSCTX_INPROC_SERVER,
17           shell.IID_IShellLink,
18       )
19       persistFile = shellLink.QueryInterface(pythoncom.IID_IPersistFile)
20       persistFile.Load(fname, STGM_READ)
21       shellLink.Resolve(0, shell.SLR_ANY_MATCH | shell.SLR_NO_UI)
22       fname, findData = shellLink.GetPath(0)
23       print("Filename:", fname, ", UNC=", shellLink.GetPath(shell.SLGP_UNCPRIORITY)[0])
24       print("Description:", shellLink.GetDescription())
25       print("Working Directory:", shellLink.GetWorkingDirectory())
26       print("Icon:", shellLink.GetIconLocation())
27   
28   
29   def FavDumper(nothing, path, names):
30       # called by os.path.walk
31       for name in names:
32           print(name, end=" ")
33           try:
34               DumpLink(name)
35           except pythoncom.com_error:
36               print(" - not a link")

39   def DumpFavorites():
40       favfold = str(shell.SHGetSpecialFolderPath(0, shellcon.CSIDL_FAVORITES))
41       print("Your favourites are at", favfold)
42       os.path.walk(favfold, FavDumper, None)

45   if __name__ == "__main__":
46       if len(sys.argv) > 1:
47           for fspec in sys.argv[1:]:
48               files = glob.glob(fspec)
49               if files:
50                   for file in files:
51                       print(file)
52                       DumpLink(file)
53                       print()
54               else:
55                   print("Can not find", fspec)
56       else:
57           print("Dumping your favorites folder!")
58           DumpFavorites()
